[docs]class PointSet(GlyphBase):
    """
    Add a set of x, y points to a figure (scatter plot)

    Serves as baseclass for point-based elements.

    Parameters
    ----------
    marker : str, optional
        The marker to use for plotting, by default 'circle'
    data : dict[str, Sequence], optional
        Dictionary containing at least the keys :code:`'cx', 'cy'` which are used
        to define the point positions. Values are equal-length sequences of, numeric
        in the case of :code:`'cx', 'cy'`, which can be used to defined properties
        on each point. If None, and empty glyph is created.
    **kwargs : dict, optional
        Optional :code:`kwargs` passed to the Scatter constructor, can be
        used to set style options, in particular can be used to map extra columns
        in :code:`data` to properties, allowing per-point style options.
    """
    def __init__(self, *args, **kwargs):
        self._selectors = []
        super().__init__(*args, **kwargs)

    @staticmethod
    def make_point_tool(renderers=None):
        if renderers is None:
            renderers = []
        return PointDrawTool(name='Point Draw/Edit',
                             description='Draw points on figure',
                             renderers=renderers,
                             add=True,
                             drag=True,
                             custom_icon=None,
                             empty_value=GlyphBase._default_fill_value)

    @staticmethod
    def editable_glyphs():
        return (Scatter,)

    @staticmethod
    def tool_label():
        return 'points'

    @property
    def default_fill_values(self):
        return {'r0': 1., 'r1': 2., 'label': 'undefined'}

    @staticmethod
    def make_select_tool(renderers=None):
        if renderers is None:
            renderers = []
        tool = LassoSelectTool(select_every_mousemove=False,
                               name='Select points',
                               description='Select points by dragging',
                               renderers=renderers)
        return tool

    @staticmethod
    def select_tool_name():
        return 'point_select_tool'

    @property
    def cx(self):
        try:
            return self._cx_key
        except AttributeError:
            return 'cx'

    @cx.setter
    def cx(self, key: str):

        assert not hasattr(self, '_cx_key')
        self._cx_key = key

    @property
    def cy(self):
        try:
            return self._cy_key
        except AttributeError:

    @cy.setter

        assert isinstance(key, str)

    def build_figure(self,
                     *,
                     marker: str = 'circle',
                     data: dict[str, Sequence] = None,
                     source=None,
                     keys=None,
                     **kwargs):
        if keys is not None:
            self.cy, self.cx = keys
        if source is not None:
            assert isinstance(source, ColumnDataSource)
            self.cds = source
        else:
            if data is None:
                data = {'cx': [], 'cy': []}
            assert self.verify_data(data)
            self.cds = ColumnDataSource(data)
        glyph = Scatter(marker=marker,
                        x=self.cx,
                        y=self.cy,
                        line_color=kwargs.pop('line_color', None),
                        fill_color=kwargs.pop('fill_color', 'red'),
                        fill_alpha=kwargs.pop('fill_alpha', 1.),
                        size=kwargs.pop('size', 10),
                        **kwargs)
        self.add_glyph(self.cds, glyph)

    @staticmethod
    def verify_data(data):
        assert isinstance(data, dict)
        assert 'cx' in data.keys()
        assert 'cy' in data.keys()
        assert all(isinstance(d, (list, np.ndarray)) for d in data.values())
        assert len({len(d) for d in data.values()}) == 1
        return True


[docs]    def replace_points(self, data: dict[str, Sequence], idx_label: bool = False):
        """
        Replace the data backing the plot and trigger a plot update

        Parameters
        ----------
        data : dict[str, Sequence]
            Dictionary containing the same keys as the current data source
        idx_label : bool, optional
            If True, add a :code:`'label'` column to the data source
            consisting of the index of the point in the data, which
            can be used to display the index of each point on the plot.
            Only possible if :code:`'label'` is not already in data.
        """
        assert self.cds is not None
        if idx_label:
            assert 'label' not in data.keys()
            labels = [str(s) for s in range(len(data['cx']))]
            data['label'] = labels
        assert self.verify_data(data)
        assert data.keys() == self.cds.data.keys()
        options = self._get_options(data=data)
        for select in self._selectors:
            select.options = options
        self.cds.data = data



[docs]    def add_labels(self, labels: Optional[Sequence[str]] = None, as_idx: bool = True, **kwargs):
        """
        Add text labels each point in the data source and display on the figure

        Parameters
        ----------
        labels : Optional[Sequence[str]], optional
            Custom labels, must match length of current data source.
            By default None, in which case pre-defined labels are used
            or if :code:`as_idx == True` numeric index labels are used
            instead.
        as_idx : bool, optional
            If :code:`labels is None` or labels not already defined on the data
            apply integer index labels to each point, by default True
        **kwargs : dict, optional
            Optional :code:`kwargs` passed to the Text constructor, which can be
            used to set style options, in particular can be used to map extra columns
            in :code:`data` to text properties, allowing per-point label options.
        """
        if labels is None:
            if not as_idx:
                assert 'label' in self.cds.data.keys()
            else:
                labels = [str(s) for s in range(self.num_points)]
                self.cds.data.update({'label': labels})
        else:
            assert len(labels) == self.num_points
            self.cds.data.update({'label': [str(s) for s in labels]})

        glyph = Text(x='cx',
                     y='cy',
                     text='label',
                     text_color=kwargs.pop('text_color', 'red'),
                     x_offset=kwargs.pop('x_offset', 5),
                     y_offset=kwargs.pop('y_offset', 5),
                     **kwargs)
        self.add_glyph(self.cds, glyph)


    def _get_options(self, data=None):
        if data is None:
            data = self.cds.data
        if 'label' in data:
            options = [str(s) for s in data['label']]
        else:
            options = [str(s) for s in range(len(data['cx']))]
        return options


[docs]    def make_point_select(self, init_idx: int = 0, name: str = 'Point select', **kwargs) -> Select:
        """
        Make a selection box to select between points

        Parameters
        ----------
        init_idx : int, optional
            The index of the initial point to select, by default 0
        name : str, optional
            The label applied to the selection box, by default 'Point select'
        **kwargs : dict, optional
            Optional :code:`kwargs` passed to the Select constructor, which can be
            used to set style options.

        Returns
        -------
        Select
            The selection box
        """
        options = self._get_options()
        try:
            value = options[init_idx]
        except IndexError:
            value = None
        self._selectors.append(Select(title=name,
                                      value=value,
                                      options=options,
                                      **kwargs))
        cb = CustomJS(args={'selector': self._selectors[-1]},
                      code=self._update_selector_js())
        self.cds.js_on_change('data', cb)
        return self._selectors[-1]


    @staticmethod
    def _update_selector_js():
        return '''
if (cb_obj.length == selector.options.length){
    return
}
selector.options = [...Array(cb_obj.length).keys()].map(x => x.toString())
if (cb_obj.length == 0){
    selector.value = ''
    return
}
if (!selector.options.includes(selector.value)){
    selector.value = selector.options[0]
}
'''

        Returns

        int
            The number of points

[docs]    def add_disk(self, r0: Union[float, Sequence[float]], **kwargs) -> 'GlyphRenderer':
        """
        Add a circle annotation around each point with radius
        in data units specified by r0

        Parameters
        ----------
        r0 : Union[float, Sequence[float]]
            Either a single radius or a sequence of radii matching
            the number of defined points
        **kwargs : dict, optional
            Optional :code:`kwargs` passed to the Circle constructor, which can be
            used to set style options, in particular can be used to map extra columns
            in :code:`data` to text properties, allowing per-circle options.

        Returns
        -------
        GlyphRenderer
            Bokeh GlyphRenderer of the circles
        """
        if not isinstance(r0, (tuple, list, np.ndarray)):
            r0 = [r0] * self.num_points
        assert len(r0) == self.num_points

        self.cds.data['r0'] = r0
        glyph = Circle(x='cx',
                       y='cy',
                       radius='r0',
                       radius_units='data',
                       line_color=kwargs.pop('line_color', 'red'),
                       line_width=kwargs.pop('line_width', 2),
                       fill_color=kwargs.pop('fill_color', 'red'),
                       fill_alpha=kwargs.pop('fill_alpha', 0.5),
                       **kwargs)
        return self.add_glyph(self.cds, glyph)



[docs]    def add_annulus(self,
                    r0: Union[float, Sequence[float]],
                    r1: Union[float, Sequence[float]],
                    **kwargs):
        """
        Add an annulus annotation around each point with radii
        in data units specified by r0 and r1

        Parameters
        ----------
        r0 : Union[float, Sequence[float]]
            Either a single inner radius or a sequence of inner radii matching
            the number of defined points
        r0 : Union[float, Sequence[float]]
            Either a single outer radius or a sequence of outer radii matching
            the number of defined points
        **kwargs : dict, optional
            Optional :code:`kwargs` passed to the Annulus constructor, which can be
            used to set style options, in particular can be used to map extra columns
            in :code:`data` to text properties, allowing per-annulus options.

        Returns
        -------
        GlyphRenderer
            Bokeh GlyphRenderer of the annuli
        """
        if not isinstance(r0, (tuple, list, np.ndarray)):
            r0 = [r0] * self.num_points
            r1 = [r1] * self.num_points
        assert len(r0) == self.num_points
        assert len(r1) == self.num_points

        self.cds.data.update({'r0': r0, 'r1': r1})
        glyph = Annulus(x='cx',
                        y='cy',
                        inner_radius='r0',
                        outer_radius='r1',
                        inner_radius_units='data',
                        outer_radius_units='data',
                        line_color=kwargs.pop('line_color', 'red'),
                        line_width=kwargs.pop('line_width', 2),
                        fill_color=kwargs.pop('fill_color', 'red'),
                        fill_alpha=kwargs.pop('fill_alpha', 0.5),
                        **kwargs)
        return self.add_glyph(self.cds, glyph)


    @staticmethod
    def duplicate_value(data, key, refkey):
        assert key in data.keys()
        assert refkey in data.keys()
        assert isinstance(data[refkey], (tuple, list, np.ndarray))
        if not isinstance(data[key], (tuple, list, np.ndarray)):
            data[key] = [data[key]] * len(data[refkey])
        return data

    @property
    def annotation_key(self):
        return 'point'




[docs]class DiskSet(PointSet):
    """
    Add a set of points with an associated radius and Circle glyph

    See :class:`~Pointset` for the remainder of the signature.

    Parameters
    ----------
    disk_kwargs : dict[str, Sequence[Any]], optional
        Optional dict of kwargs passed to the Circle constructor, which can be
        used to set style options.
    **kwargs : dict[str, Sequence[Any]], optional
        Optional :code:`kwargs` passed to the PointSet constructor, which can be
        used to set style options, in particular can be used to map extra columns
        in :code:`data` to text properties, allowing per-point options.
    """
    def build_figure(self, *args,
                     data: dict[str, Sequence[Any]] = None,
                     disk_kwargs: dict[str, Sequence[Any]] = None,
                     extra_keys: list[str] = None,
                     **kwargs):
        if data is None:
            data = {'r0': [], 'cx': [], 'cy': []}
            if extra_keys:
                data.update({k: [] for k in extra_keys})
        data = self.duplicate_value(data, 'r0', 'cx')

        super().build_figure(*args, data=data, **kwargs)
        if disk_kwargs is None:
            disk_kwargs = {}
        self.add_disk(r0=data['r0'], **disk_kwargs)

    def link_callback(self, slider=None):
        if slider is None:
            slider = self.slider
        assert slider is not None

        args = {'slider': slider,
                'cds': self.cds}

        select_cb = CustomJS(args=args,
                             code=self._select_disk_slider_js())
        self.cds.selected.js_on_change('indices', select_cb)

        slider_cb = CustomJS(args=args,
                             code=self._slider_change_js())
        self.slider.js_on_change('value', slider_cb)

    def _select_disk_slider_js(self):
        return '''
if (cb_obj.indices.length >= 1){
    slider.disabled = false;
    const r_values = cb_obj.indices.map(x => cds.data.r0[x]);
    slider.value = Math.min(...r_values);
} else {
    slider.disabled = true;
}
'''

    def _slider_change_js(self):
        return '''
const selected = cds.selected.indices
var idx = -1
if (selected.length == 0) {
    return;
} else {
    selected.forEach(idx => cds.data.r0[idx] = slider.value);
    cds.change.emit()
    cds.data = cds.data;
    cds.properties.data.change.emit();
}
