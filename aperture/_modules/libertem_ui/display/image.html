    """
    Adds a Bokeh Image glyph to a figure and provides convenience methods on it

    Parameters
    ----------
    xar : xa.DataArray
        The array to display as the image, which would normally be supplied
        by the constructor :meth:`~BokehImage.from_numpy`.
    fig : Optional[BokehFigure], optional
        The figure on which to place the image, by default None for which
        a figure will be created when first accessed. When constructed via
        the method on BokehFigure, fig will be automatically set.
    anchor_xy : tuple[float, float], optional
        The anchor at which to set the image glyph, by default (0., 0.)
        This is applied in addition to the :code:`cls._px_offset` attribute
        and allows overlaying a BokehImage onto other glyphs with a non-zero
        origin (otherwise 0,0 is implicit in the class construction).

    Attributes
    ----------
    _px_offset: float
        -0.5 by default, used to offset the image origin by a half pixel
        so that the pixels as-displayed are centred on a whole coordinate
        and not anchored at their top-left. Set this to zero to anchor pixels
        at their top-left.
    """
    _px_offset: float = -0.5

    def __init__(self, *,
                 xar: 'xa.DataArray',
                 fig: Optional['BokehFigure'] = None,
                 anchor_xy: tuple[float, float] = (0., 0.),
                 **kwargs):
        self.fig = fig
        self.set_anchor(x=anchor_xy[0], y=anchor_xy[1])
        self._glyph = None

        assert isinstance(xar, get_xa().DataArray)
        self.raw_image = xar
        assert self.raw_image.ndim == 2, 'multichannel images not yet supported'

        super().__init__(fig=fig, editable=False, **kwargs)


[docs]    @classmethod
    def from_numpy(cls,
                   *,
                   array: np.ndarray,
                   dimensions: Optional[list[dict[str, Union[str, float]]]] = None,
                   attrs: dict[str, Any] = None,
                   fig: Optional['BokehFigure'] = None,
                   **kwargs) -> 'BokehImage':
        """
        Construct a :class:`~BokehImage` from a 2-D numpy array

        The internal data structure used is an :code:`xarray.DataArray`, which is constructed by
        this method before initializing the class. The attributes which can be set
        are used by the class to define, at the present, coordinate systems for
        each array axis and labels (titles, units).

        By default the class sets a pixel-aligned coordinate system for the primary
        axes (:code:`'_y', '_x'`), which is used for all annotations defined on the image.
        To specify a secondary, dimensioned coordinate system supply the :code:`dimensions`
        argument to this class with one of the following formats:

        **Dimension specification**

        dimension['name'] : str, required
            A unique name for the dimension, used in the axis label
        dimension['flipped'] : bool, optional
            Whether the dimension should be flipped when displayed, False by default
        dimension['unit'] : str, optional
            A unit label for the dimension, applied to the axis label
        dimension[...] : str, float
            See :meth:`~BokehImage.build_coord_range` for numerical dimension
            specification.


        Similarly certain attributes can be specficied on the array:

        **Attribute specification**

        attrs['title'] : str, optional
            A title for the array
        attrs['array_unit'] : str, optional
            A unit for the array elements, appended to the title and colorbar


        Parameters
        ----------
        array : np.ndarray
            A 2-dimensional numpy array
        dimensions : Optional[list[dict[str, Union[str, float]]]], optional
            Coordinate system definition for each dimension of the array, by default None,
            in which case only the pixel coordinate system is created. If defined
            should be a 2-list of dicts of coordinate attributes as detailed above.
        attrs : dict[str, Any], optional
            Dict of extra attributes to attach to the image, by default None,
            as detailed above.
        fig : Optional[BokehFigure], optional
            An existing BokehFigure class to attach to, by default None,
            usually set automatically if constructed by the :code:`BokehFigure.add_image`
            method.
        **kwargs : dict, optional
            Optional :code:`kwargs` passed to the BokehImage init

        Returns
        -------
        BokehImage
            The image constructed from the array
        """
        if isinstance(array, get_xa().DataArray):
            raise ValueError(f'Try instantiating the class directly, got {type(array)}')
        cls.check_nparray(array)
        base_dims, attrs, coords = cls._base_xa_config(array, attrs=attrs)
        if dimensions is not None:
            assert len(dimensions) == array.ndim
            attrs['dimension_def'] = {}
            for dimension, base_dim, dim_size in zip(dimensions, base_dims, array.shape):
                dim_name = dimension['name']
                assert isinstance(dimension, dict)
                if dim_name in base_dims:
                    raise ValueError(f'Dimension names: {base_dims} are reserved')
                coordinate_sequence = cls.build_coord_range(dimension, dim_size)
                # Making and assigning the sequence is not actually necessary
                # with the current use of dimensioned coordinates, and might
                # actually be a perf issue as we interpolate the extra coords each time
                # we index into the array
                # coords[dim_name] = (base_dim, coordinate_sequence)
                assert coordinate_sequence[0] != coordinate_sequence[-1], \
                    ('Coordinate defintion must define a range got '
                     f'll: {coordinate_sequence[0]} ul: {coordinate_sequence[-1]}')
                attrs['dimension_def'][base_dim] = {'name': dim_name,
                                                    'lower': coordinate_sequence[0],
                                                    'upper': coordinate_sequence[-1]}
                if 'flipped' in dimension.keys():
                    attrs['dimension_def'][base_dim]['flipped'] = dimension['flipped']
                if 'unit' in dimension.keys():
                    attrs['units'][dim_name] = dimension['unit']

        xar = get_xa().DataArray(data=array, dims=base_dims, coords=coords, attrs=attrs)
        return cls(xar=xar, fig=fig, **kwargs)


    @staticmethod
    def _base_xa_config(array, attrs=None):
        if attrs is None:
            attrs = {}
        base_dims = ['_y', '_x']
        attrs['units'] = {k: 'px' for k in base_dims}
        coords = {k: np.arange(s) for k, s in zip(base_dims, array.shape)}
        return base_dims, attrs, coords

    def build_coord_range(dimension_def: dict[str, Any], num_el: int) -> np.ndarray:
        """
        Create a sequence of coordinate values for a dimension of an array
        :code:`dimension_def` can have one of two specifications.

        This function is not intended to be called by the user,
        but is documented to support the constructor :meth:`~BokehFigure.from_numpy`

        **Spec 1**

        dimension_def['px_size'], float
            The step size per pixel
        dimension_def['origin'], float, optional, default is 0.
            The origin of the coordinate sequence


        **Spec 2**

        dimension_def['lower'], float
            The inclusive lower bound of the coordinate sequence
        dimension_def['upper'], float
            The inclusive upper bound of the coordinate sequence


        A further optional paramter is :code:`dimension_def['dtype']`, default
        :code:`np.float32` which specifies the dtype of the coordinate
        system.

        Parameters
        ----------
        dimension_def : dict[str, Any]
            A dimension def dictionary, described above
        num_el : int
            Number of elements along this axis of the array

        Returns
        -------
        np.ndarray
            The sequence of coordinate values defined by :code:`dimension_def`
            for a dimension of length :code:`num_el`
        """

        dtype = dimension_def.get('dtype', np.float32)
        # if 'coord_sequence' in dimension_def:
        #     assert len(dimension_def['coord_sequence']) == num_el
        #     return np.asarray(dimension_def['coord_sequence'])
        if 'px_size' in dimension_def:
            origin = dimension_def.get('origin', 0.)
            stepsize = dimension_def['px_size']
            assert stepsize != 0.
            if num_el > 1:
                return (origin + (np.cumsum(np.ones((num_el - 1,))) - 1) * stepsize).astype(dtype)
            else:
                return np.asarray((origin - (stepsize / 2), origin + (stepsize / 2))).astype(dtype)
        elif all(k in dimension_def for k in ['lower', 'upper']):
            if num_el == 1:
                raise ValueError('Cannot specify lower/upper for single-element dimension')
            ll = dimension_def['lower']
            ul = dimension_def['upper']
            return np.linspace(ll, ul, num=num_el, endpoint=True).astype(dtype)
        else:
            raise ValueError(f'Unrecognized dimension spec {dimension_def.keys()}')

    @raw_image.setter
    def raw_image(self, array: 'xa.DataArray'):
        """
        Sets the raw image to the array supplied
        See :meth:`~BokehImage.update_raw_image`.
        Should not be called directly by the user.

        When this is set we automatically trigger an update
        to the self.display_image property and push updates
        to the plot (if appropriate)

        Parameters
        ----------
        array : xa.DataArray
            DataArray to set as the current raw image


        :meta private:
        """
        xa = get_xa()
        if not isinstance(array, xa.DataArray):
            base_dims, attrs, coords = self._base_xa_config(array, attrs=self.raw_image.attrs)
            array = xa.DataArray(data=array, dims=base_dims, coords=coords, attrs=attrs)
        self._raw_image = array
        self.update_display_image(force=True)

    @staticmethod
    def check_nparray(array):
        assert isinstance(array, np.ndarray)
        assert array.ndim == 2, 'No support for multi-channel images yet'
        assert all(s > 0 for s in array.shape)
        assert not np.issubdtype(array.dtype, complex), 'No support for complex images yet'
        return True

    @staticmethod
    def convert_dtype(array):
        if np.issubdtype(array.dtype, np.floating) and array.itemsize != 4:
            return array.astype(np.float32)
        return array


[docs]    def update_raw_image(self, array: np.ndarray, force: bool = False,
                         maxrate: int = 5,
                         fix_clims: Union[bool, tuple[float, float]] = False):
        """
        Update the current displayed image with the supplied array

        This is the user-facing endpoint for triggering image updates

        The supplied array must have the same shape as the array
        on which the class was initialized. If a change of shape is desired
        the user can try the :meth:`~BokehImage.raw_image` setter, but this
        is generally unsupported.

        Parameters
        ----------
        array : np.ndarray
            The array to update with
        force : bool, optional
            Set true to force an update to the plot even if the updates
            are being rate limited, otherwise, by default False
        maxrate : int, optional
            The maximum number of times per-second to update, by default 5.
            If this method is called faster than this then the internal data
            of the image is updated but updates are not pushed to the browser.
            Call :meth:`~BokehImage.refresh_display` after fast updating
            in order to ensure the browser version is up-to-date.
        fix_clims : Union[bool, tuple[float, float]], optional
            Fix the image color limits when updating, by default False.
            Can either be a boolean True, in which case the previous color
            limits are maintained, or a 2-tuple of floats which are used
            to force a particular color limits.
            This is mainly interesting when the update rate is expected to be
            high, as it prevents the need to calculate the limits and push
            updates to the browser (which then has to do the mapping).
            Again call :meth:`~BokehImage.refresh_display` to ensure
            everything is in sync after fast updating.
        """
        self.check_nparray(array)
        assert self.raw_image is not None, 'Cannot update raw image before init'
        assert self.raw_image.shape == array.shape, 'Use raw_image setter if changing shape'
        self.raw_image.data = array
        self.update_display_image(force=force, maxrate=maxrate, fix_clims=fix_clims)


    def update_display_image(self, force=False, maxrate=5, fix_clims=False):
        """
        Internal function to convert the recieved raw image into a format
        suitable for display, most notably down-converting np.float64
        images into np.float32 to reduce transport size. Params as
        with :meth:`~BokehImage.update_raw_image`.

        :meta private:
        """
        # This is here so we can do dtype conversion to reduce image size
        display_array = self.convert_dtype(self.raw_image.data)
        if force or self.display_image is None:
            self._display_image = self.raw_image.copy(data=display_array)
            self.update_array_limits()
            self._update_display_hook()
        else:
            # This is the normal path for an update with an equivalent
            # array (shape + dtype + coordinates)
            self._display_image.data = display_array

        if force or (not self.rate_limit(maxrate)):
            self._force_update(fix_clims)

    def _update_display_hook(self):
        # Allows datashaded class to inject calls
        pass

    def _force_update(self, fix_clims=False):
        if self.has_im():
            if isinstance(fix_clims, (list, tuple, np.ndarray)):
                self.update_array_limits(lims=fix_clims)
            elif not fix_clims:
                self.update_array_limits()
            self.update_cds()

    def update_array_limits(self, lims=None):
        """
        Updates the cache of min/max values of the current array
        Computes on the whole array, not a reduced size version,
        so for very large arrays this can be quite slow!

        Assignment to self.display_minmax causes the values to be synced
        to the browser and the colorbar and slider to be updated via
        Javascript callbacks

        For maximum speed this update can be skipped by passing fix_clims=True
        when calling self.update_raw_image(). For manually fixed colorlimits
        pass a 2-tuple of floats which will be used rather than calculating
        the values from the array itself.

        Testing various accelerations for this (numba, ufunclab) show that a double
        pass with numpy is fastest on a 1k x 1k np.float32 array
        Here we are optimistic and first assume no NaNs, as the nanmin/nanmax way
        seems about 5x slower for this same 1k array

        :meta private:
        """
        if lims is not None:
            self.display_minmax = lims
        else:
            self.display_minmax = self.calc_minmax(self.display_image.data)

    @staticmethod
    def calc_minmax(array):
        data = array.ravel()
        minmax = np.min(data), np.max(data)
        if np.isnan(minmax).any():
            minmax = np.nanmin(data), np.nanmax(data)
        return minmax

    def update_tick(self) -> None:
        """
        Signal that we did an update

        :meta private:
        """
        self._last_update = time.time()

    def rate_limit(self, maxrate: int) -> bool:
        """
        Computes whether we should rate limit

        :meta private:
        """
        if maxrate == 0:
            return False
        assert maxrate > 0
        now = time.time()
        if (now - self.last_update) > (1. / maxrate):
            self.update_tick()
            return False
        else:
            return True

    @display_minmax.setter
    def display_minmax(self, val: tuple[float, float]):
        """
        Set the current display image min/max values

        This method also syncs the min/max values to the
        browser copy so that sliders and colorbars can respond

        :meta private:
        """
        assert len(val) == 2
        if val[0] == val[1]:
            # should not ever trigger if val == (nan, nan)
            assert not any(np.isnan(v) for v in val)
            # This avoids various cases where low == high causes errors
            # By default use 1% of val, else 1e-5 as a fallback
            offset = val[0] * 0.01 if val[0] else 1e-5
            val = (val[0], val[0] + offset)
        self._disp_minmax = val
        update = {'_low': [val[0]], '_high': [val[1]]}
        com_val = self._combined_minmax()
        new_datadict = {'low': [com_val[0]], 'high': [com_val[1]], **update}
        if any(np.isnan(v[0]) for v in new_datadict.values()):
            new_datadict = {'low': [0.], 'high': [1.]}
        self._update_cds_datadict(self.display_meta_cds, new_datadict)

    def continuous_bounds(self, apply_px_offset=True):
        """
        The bounds of the full image in continuous coordinates
        Returns position of outer edges of pixels as-drawn

        :meta private:
        """
        h, w = self.array_hw()
        if apply_px_offset:
            anchor_x = anchor_y = self._px_offset
        else:
            anchor_x = anchor_y = 0.
        anchor_x, anchor_y = self.shift_anchor(anchor_x, anchor_y)
        return anchor_x, anchor_y, anchor_x + w, anchor_y + h

    def array_bounds(self):
        """
        The array bounds in integer pixels

        :meta private:
        """
        h, w = self.array_hw()
        return 0, 0, w, h

    def array_hw(self):
        """
        Returns full array shape (h, w)

        :meta private:
        """
        return self.display_image.shape

    def array_aspect(self):
        """
        display_image array aspect ratio (w / h)

        :meta private:
        """
        # matches bokeh convention
        h, w = self.array_hw()
        aspect = w / h
        assert aspect == self.fig.aspect_ratio_def(height=h, width=w), 'Implementation changed?'
        return aspect

    def current_cds_dims(self):
        """
        Current geometry of the glyph displaying the image

        :meta private:
        """
        if self.cds is None:
            return None
        return {k: self.cds.data[k] for k in ['x', 'y', 'dw', 'dh']}

    def update_cds(self, new_datadict: dict[str, Any] = None, force_full: bool = False):
        """
        Update the datasource with the full array

        If the image coordinates have not changed, just update the array data
        else update the whole CDS. If this is the first time the CDS
        is being populated, this function acts as the constructor.
        Should be called automatically each time the image data is updated

        Parameters
        ----------
        new_datadict : dict[str, Any], optional
            If supplied, update with this specific dictionary, by default None
            in which case a full image dictionary is generated from the array
        force_full : bool, optional
            Force use of a full array dictionary, by default False


        :meta private:
        """
        if force_full or new_datadict is None:
            new_datadict = self.get_full_datadict()

        if self.cds is not None:
            self._update_cds_datadict(self.cds, new_datadict)
        else:
            self.cds = ColumnDataSource(new_datadict)

    def get_full_datadict(self, image_data=None):
        """
        Provides Bokeh CDS dictionary based on continuous coordinates
        left and top are the anchor point accounting for any self._px_offset
        (h, w) are the actual number of elements in each dimension

        :meta private:
        """
        l, t, _, _ = self.continuous_bounds()
        h, w = self.array_hw()
        if image_data is None:
            image_data = self.display_image.data
        return self.get_datadict(image_data, l, t, h, w)

    @staticmethod
    def get_datadict(image_array, left, top, height, width):
        """
        Return a correctly formatted datadict based on image geometry and data
        Formats the anchor point to be plotted with an inverted y-axis
        matching numpy/matplotlib convention for arrays/images. As a result
        also flips the array data to match.

        :meta private:
        """
        return {'x': [left], 'y': [top + height], 'dw': [width], 'dh': [height],
                'image': [image_array[::-1, :]]}

        try:

        except AttributeError:
            return False


[docs]    def add_dimensioned_axes(self, hor_pos: str = 'above', ver_pos: str = 'right'):
        """
        Add dimensioned secondary axes to the figure using the coordinate
        system defintion passed to the constructor :meth:`~BokehImage.from_numpy`.

        The two position arguments are both Bokeh position spec strings.

        Will do nothing if no coordinate system was defined at construction time.

        Parameters
        ----------
        hor_pos : str, optional
            Position to add the horizontal axis, by default 'above'
        ver_pos : str, optional
            Position to add the vertical axis, by default 'right'
        """
        if not self.has_dimensions():
            return
        position_mapping = {'_y': ver_pos, '_x': hor_pos}
        units = self.display_image.attrs['units']
        for base_dim, position in position_mapping.items():
            dimension_def = self.display_image.attrs['dimension_def'][base_dim]
            lower = dimension_def['lower']
            upper = dimension_def['upper']
            name = dimension_def['name']
            label = name + f' ({units.get(name, "-")})'
            flipped = True if position in ['right', 'left'] else False
            flipped = dimension_def.get('flipped', flipped)
            self.fig._add_extra_axis(position, name, lower, upper,
                                     label=label,
                                     flipped=flipped)

                     palette: Optional[str] = None):
        """
        Set the :code:`BokehFigure` to have appropriate axes and
        dimensions for the `BokehImage` object, notably inverting the
        y-axis and setting a specific data aspect. Generally this
        should only be called one per `BokehFigure`, and should raise an error
        if done multiple times.

        When constructing this class via the `BokehFigure.add_image` method
        the :code:`setup_figure` method is automatically called if it is the first
        time an image has been added to this figure.

        Parameters
        ----------
        data_aspect : float, optional
            Aspect ratio of the data as drawn, by default 1 in which case
            pixels will be square.
        frame_width : Optional[int], optional
            Forwarded to :meth:`BokehFigure.set_frame_aspect`, by default None
        frame_height : Optional[int], optional
            Forwarded to :meth:`BokehFigure.set_frame_aspect`, by default None
        palette : Optional[str], optional
            The color palette to use for the image.
            Forwarded to :code:`set_color_mapper`, by default None
        """
        # Auto set data_aspect based on dimensioned coordinates ???
        self.fig.set_match_aspect(True)
        self.fig.set_frame_aspect(self.array_aspect() * data_aspect,
                                  frame_width=frame_width,
                                  frame_height=frame_height)
        self.set_color_mapper(palette=palette)
        self.force_box_zoom_aspect()
        self.set_axes_ranges()
        self.set_axis_labels()
        self.set_title()
        self.setup_cds()


    def setup_cds(self):
        self.update_cds(force_full=True)

    def _default_axis_label(self, idx):
        base_dim = self.display_image.dims[idx]
        unit = self.display_image.attrs['units'].get(base_dim, '-')
        return f'{base_dim.replace("_", "")} ({unit})'

        """
        Set primary axis labels on the figure

        The values are by default x (px) / y (px) but can be overridden here

            x = self._default_axis_label(1)
        self.fig.set_axis_labels(x=x, y=y)


    def force_box_zoom_aspect(self):
        # Sets box zoom aspect ratio to fixed
        # This function should probably be a method of BokehFigure
        # Also refactor to work with bkfig._tools
        is_zoom = [isinstance(t, BoxZoomTool) for t in self.fig.fig.tools]
        if any(is_zoom):
            if sum(is_zoom) > 1:
                raise RuntimeError('More than one BoxZoomTool??')
            true_idx = is_zoom.index(True)
            self.fig.fig.tools[true_idx].match_aspect = True


[docs]    def set_title(self, title: Optional[str] = None):
        """
        Sets the figure title. If title is not supplied
        the title is created from the :code:`attrs['title']`
        and :code:`attrs['array_unit']` attributes on the
        underlying array, which were supplied to the constructor.

        Parameters
        ----------
        title : Optional[str], optional
            The title to set, by default None in which case
            the title is inferred from the array attributes
        """
        if title is None:
            if 'title' in self.display_image.attrs:
                title = self.display_image.attrs['title']
            if 'array_unit' in self.display_image.attrs:
                title = title + f' ({self.display_image.attrs["array_unit"]})'
        if title is not None:
            self.fig.set_title(title)


    @property
    def anchor_offset(self):
        try:
            return self._anchor_offset
        except AttributeError:
            self._anchor_offset = (0., 0.)
            return self.anchor_offset

    def set_anchor(self, *, x: float, y: float):
        """
        Sets the image anchor point

        :meta private:
        """
        self._anchor_offset = (x, y)

    def shift_anchor(self, left, top) -> tuple[float]:
        """
        Applies the forward translation to get the offset anchor

        :meta private:
        """
        dx, dy = self.anchor_offset
        return left + dx, top + dy

    def unshift_anchor(self, left, top) -> tuple[float]:
        """
        Applies the reverse translation to get the non-offset anchor

        :meta private:
        """
        dx, dy = self.anchor_offset
        return left - dx, top - dy


[docs]    def centre_anchor(self, refresh=True):
        """
        Shifts the image anchor to place the image with (0, 0)
        in axis units in the centre of the glyph
        Essentially nullifies cls._px_offset to do this
        with consistent behaviour on odd/even dimensions

        Does not currently update the the axes ranges!!!
        """
        h, w = self.array_hw()
        x_shift = -self._px_offset - (w / 2)
        y_shift = -self._px_offset - (h / 2)
        self.set_anchor(x=x_shift, y=y_shift)
        if refresh:
            self.refresh_display()


    def reset_anchor(self, refresh=True):
        if self.anchor_offset == (0., 0.):
            return
        self.set_anchor(x=0., y=0.)
        if refresh:
            self.refresh_display()


[docs]    def current_centre_xy(self) -> tuple[float, float]:
        """
        Get the current centre of the image glyph,
        taking the anchor offset into account

        Returns
        -------
        tuple[float, float]
            the current centre of the image glyph
        """
        left, top, right, bottom = self.continuous_bounds(apply_px_offset=False)
        return (sum([left, right]) / 2., sum([top, bottom]) / 2.)
