[docs]class Rectangles(GlyphBase):
    """
    Add a Rectangle glyph to the image. When built with :code:`allow_edit=True`
    the user can interactively add and remove rectangles using the automatically
    added tool.

    Parameters
    ----------
    rectangles : list[tuple[float, float, float, float]], optional
        List of inital rectangles in a :code:`cx, cy, w, h` format.
    **kwargs : dict, optional
        Optional :code:`kwargs` passed to the Bokeh Rect constructor, can be
        used for to set style options.
    """
    @staticmethod
    def make_rectangle_tool(renderers=None):
        if renderers is None:
            renderers = []
        return BoxEditTool(name='Rectangle Draw/Edit',
                           description='Draw rectangles on figure',
                           renderers=renderers,
                           custom_icon=None,
                           empty_value=GlyphBase._default_fill_value)

    @staticmethod
    def editable_glyphs():
        return (Rect,)

    @staticmethod
    def tool_label():
        return 'boxes'

    def build_figure(self, *, rectangles: list[tuple[float, float, float, float]] = None, **kwargs):
        self.cds = ColumnDataSource(self._empty_cds_dict())
        if rectangles is not None:
            self.cds.data.update(self._dict_from_xywh(rectangles))

        glyph = Rect(x='cx',
                     y='cy',
                     width='w',
                     height='h',
                     fill_alpha=kwargs.pop('fill_alpha', 0.3),
                     fill_color=kwargs.pop('fill_color', 'red'),
                     line_color=kwargs.pop('line_color', 'red'),
                     line_dash=kwargs.pop('line_dash', 'dashed'),
                     **kwargs)
        self.add_glyph(self.cds, glyph)

    @staticmethod
    def _cds_dict(x, y, w, h, **other):
        return {'cx': x, 'cy': y, 'w': w, 'h': h, **other}

    def _empty_cds_dict(self):
        return self._cds_dict([], [], [], [])

    def _dict_from_xywh(self, rectangles):
        rectangles = np.asarray(rectangles).reshape(-1, 4)
        cx, cy, w, h = rectangles.T.tolist()
        return self._cds_dict(cx, cy, w, h)

    def export_data(self) -> 'pd.DataFrame':
        # Need to take absolute as sometimes Bokeh flips the sign
        df = super().export_data()
        df['w'] = df['w'].abs()
        df['h'] = df['h'].abs()
        return df

    @property
    def annotation_key(self):
        return 'rectangle'



class MultiPointsetEditBase:
    """
    Base class for (editable) multi-collections of points
    which render as sets of lines or patches

    :meta private:
    """

    def _make_tool(self, cls, name, description, icon, renderers):
        if renderers is None:
            renderers = []
        return cls(name=name,
                   description=description,
                   renderers=renderers,
                   custom_icon=icon,
                   vertex_renderer=self.get_vertex_renderer(),
                   empty_value=GlyphBase._default_fill_value)

    def make_draw_tool(self, name, description, icon=None, renderers=None):
        return self._make_tool(PolyDrawTool, name, description, icon, renderers)

    def make_edit_tool(self, renderers=None):
        name = 'Line / Poly edit'
        description = 'Edit lines and polygons on figure'
        icon = None
        return self._make_tool(PolyEditTool, name, description, icon, renderers)

    @property
    def tool_mapping(self):
        return {'multipointset_edit': self.make_edit_tool}

    def tool_labels(self):
        return tuple(self.tool_mapping.keys())

    def get_vertex_renderer(self):
        try:
            return self._vertex_renderer
        except AttributeError:
            self.make_vertex_renderer()
            return self.get_vertex_renderer()

    def make_vertex_renderer(self: 'GlyphBase', **kwargs):
        cds = ColumnDataSource({'x': [], 'y': []})
        point_glyph = Circle(fill_color=kwargs.pop('fill_color', 'red'),
                             size=kwargs.pop('size', 10),
                             **kwargs)
        self._vertex_renderer = self.fig.fig.add_glyph(cds, point_glyph)

    def limit_length(self: 'GlyphBase', n_points):
        # Need to figure out how to remove single-point Lines
        # and two-point Polygons as it's possible to end
        # up in this situation when mis-clicking. The trouble is
        # that I don't necessarily know how to distinguish between
        # a two-point poly in the process of construction, and
        # the two-point and completed error case !
        assert n_points > 1
        limit_cb = CustomJS(args={'n_points': n_points},
                            code=self._limit_length_js_cb())
        self.cds.js_on_change('data', limit_cb)

    @staticmethod
    def _limit_length_js_cb():
        return '''
if (cb_obj.data.xs == 0) {
    return
}
const offsets = cb_obj.data.xs.map(x => x.length - n_points)
var modified = false

for (const [i, offset] of offsets.entries()) {
    if (offset <= 0) {
        continue
    }
    const abs_offset = Math.abs(offset)
    cb_obj.data.xs[i] = cb_obj.data.xs[i].slice(abs_offset)
    cb_obj.data.ys[i] = cb_obj.data.ys[i].slice(abs_offset)
    modified = true
}

if (modified) {
    cb_obj.change.emit()
}
'''


class MultiPointsetGlyphBase(GlyphBase):
    def _dict_from_pointlists(self, pointlists):
        xs = []
        ys = []
        for points in pointlists:
            points = np.asarray(points).reshape(-1, 2)
            xs.append(points[:, 0])
            ys.append(points[:, 1])
        return self._cds_dict(xs, ys)

    @staticmethod
    def _cds_dict(xs, ys, **other):
        return {'xs': xs, 'ys': ys, **other}

    def _empty_cds_dict(self):
        return self._cds_dict([], [])


class Lines(MultiPointsetGlyphBase, MultiPointsetEditBase):
    def make_draw_tool(self, renderers=None):
        name = 'Line draw'
        description = 'Draw lines on figure'
        icon = line_icon()
        return super().make_draw_tool(name=name,
                                      description=description,
                                      icon=icon,
                                      renderers=renderers)

    @property
    def tool_mapping(self):
        return {'line_draw': self.make_draw_tool,
                **super().tool_mapping}

    @staticmethod
    def editable_glyphs():
        return (MultiLine,)

    def build_figure(self, *,
                     lines: list[list[tuple[float, float]]] = None,
                     limit_npt: int = 0,
                     **kwargs):
        self.cds = ColumnDataSource(self._empty_cds_dict())
        if lines is not None:
            self.cds.data.update(self._dict_from_pointlists(lines))

        glyph = MultiLine(xs='xs',
                          ys='ys',
                          line_color=kwargs.pop('line_color', 'red'),
                          line_width=kwargs.pop('line_width', 2),
                          **kwargs)
        self.add_glyph(self.cds, glyph)

        if limit_npt > 1:
            self.limit_length(limit_npt)

    @property
    def annotation_key(self):
        return 'line'

class FreePointAdd(FreeDrawBase):
    tool_name = 'Add point'
    tool_description = 'Add point by dragging around location'

    @staticmethod
    def tool_icon():
        return free_point()

    def make_freeadd_tool(self, renderers=[]):
        return self.make_free_tool(renderers=renderers)

    def build_figure(self, pointset: PointSet, **kwargs):
        super().build_figure(**kwargs)
        args = {'line_cds': self.cds, 'point_cds': pointset.cds,
                'add_value': self._default_fill_value}
        add_callback = CustomJS(args=args, code=self._add_point_js_cb())
        self.cds.js_on_change('data', add_callback)

    @staticmethod
    def tool_label():
        return 'free_point_add'

    @staticmethod
    def _add_point_js_cb():
        return '''
if (line_cds.data.xs.length == 0){
    return
}
const xvals = line_cds.data.xs[0]
const yvals = line_cds.data.ys[0]

if (xvals.length == 0){
    return
}

const sum_x = xvals.reduce((a,c) => a + c, 0);
const avg_x = sum_x / xvals.length;
const sum_y = yvals.reduce((a,c) => a + c, 0);
const avg_y = sum_y / yvals.length;

var patch = {'cx': avg_x, 'cy': avg_y}
for (const [i, colname] of point_cds.columns().entries()) {
    if (colname == 'cx' || colname == 'cy'){
        continue
    }
    patch[colname] = add_value
}

for (const [i, colname] of point_cds.columns().entries()) {
    point_cds.get_column(colname).push(patch[colname])
}

point_cds.change.emit()
point_cds.data = point_cds.data;
point_cds.properties.data.change.emit();

line_cds.clear()
'''


class FreeDrawCallback(FreeDrawBase):
    tool_description = 'Draw path to trigger callback'

    @staticmethod
    def tool_icon():
        return free_draw()

    def build_figure(self, callback, **kwargs):
        super().build_figure(**kwargs)

        def _callback_with_clear(attr, old, new):
            if any(len(v) == 0 for v in new.values()):
                return
            path = {k: v[0] for k, v in new.items()}
            callback(path)
            self.cds.data = {k: [] for k in self.cds.column_names}

        self.cds.on_change('data', _callback_with_clear)

    @staticmethod
    def tool_label():
        return 'free_draw_cb'
