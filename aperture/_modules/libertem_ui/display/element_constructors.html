[docs]    def add_image(self: 'BokehFigure',
                  autosetup: bool = True,
                  datashade: bool = True,
                  data_aspect: float = 1.,
                  frame_height: int = None,
                  frame_width: int = None,
                  **kwargs) -> BokehImage:
        """
        Construct a :class:`.image.BokehImage` on the figure

        Parameters
        ----------
        autosetup : bool, optional
            Set if the figure should be configured as an image,
            by default True, notably changing the y-axis orientation
            and setting the data aspect
        datashade : bool, optional
            Whether to allow the image to be datashaded, by default True
        data_aspect : float, optional
            The intial data aspect ratio, by default 1.
        frame_height : int, optional
            The initial frame height, by default None
        frame_width : int, optional
            The initial frame width, by default None
        **kwargs : dict, optional
            Extra keyword arguments which are passed directly to
            the :meth:`~BokehImage.build_figure` method

        Returns
        -------
        BokehImage
            The constructed image
        """
        cls = BokehImageDatashaded if datashade else BokehImage
        return self._add_image(cls,
                               autosetup=autosetup,
                               data_aspect=data_aspect,
                               frame_height=frame_height,
                               frame_width=frame_width,
                               **kwargs)


    def _add_image(self: 'BokehFigure',
                   cls: Type[BokehImage],
                   autosetup: bool = True,
                   data_aspect: float = 1.,
                   frame_height: int = None,
                   frame_width: int = None,
                   **kwargs) -> BokehImage:
        if 'array' in kwargs:
            img = cls.from_numpy(fig=self, **kwargs)
        elif 'height' in kwargs and 'width' in kwargs:
            img = cls.from_shape(fig=self, **kwargs)
        elif 'array_dict' in kwargs:
            img = cls.from_dict(fig=self, **kwargs)
        elif 'array_stack' in kwargs:
            img = cls.from_stack(fig=self, **kwargs)
        elif 'callbacks_dict' in kwargs:
            img = cls.from_callbacks(fig=self, **kwargs)
        else:
            raise TypeError('Unrecognized image constructor args')
        # We do auto-setup for the first image created on this Figure
        # Don't want this for subsequent images as it sets axis ranges/labels etc.
        if autosetup and not self._is_image_fig:
            img.setup_figure(data_aspect=data_aspect,
                             frame_height=frame_height,
                             frame_width=frame_width)
            self._is_image_fig = True
        return img

    def _add_cursor(self: 'BokehFigure', cls: Type[Cursor], *args, allow_edit=True, **kwargs):
        cursor = cls(*args, fig=self, **kwargs)
        renderers = self.get_renderers(cursor, filter_glyphs=cls.editable_glyphs())
        tool_label = Cursor.tool_label()
        if allow_edit:
            if self.has_tool(tool_label):
                self.add_renderers_to_tool(tool_label, renderers)
            else:
                tool = Cursor.make_cursor_tool(renderers=renderers)
                self.add_tool(tool_label, tool)
        cursor.add_autofill()
        return cursor

[docs]    def add_editable_disks(self: 'BokehFigure', *args, **kwargs) -> EditableDiskSet:
        """
        Add an :class:`.EditableDiskSet` object to the figure
        """
        if not kwargs.pop('allow_edit', True):
            raise ValueError('Cannot have EditableDiskSet with allow_edit = False')
        editable_pointset = self._add_pointset(EditableDiskSet, *args, allow_edit=True, **kwargs)
        renderers = self.get_renderers(editable_pointset,
                                       filter_glyphs=editable_pointset.editable_glyphs())
        tool_label = editable_pointset.select_tool_name()
        if self.has_tool(tool_label):
            self.add_renderers_to_tool(tool_label, renderers)
        else:
            tool = editable_pointset.make_select_tool(renderers=renderers)
            self.add_tool(tool_label, tool)
        return editable_pointset

    def _add_multi_pointset(self, cls, *args, allow_edit=False, **kwargs):
        element = cls(*args, fig=self, editable=allow_edit, **kwargs)
        if allow_edit:
            renderers = self.get_renderers(element, filter_glyphs=cls.editable_glyphs())
            tool_labels = element.tool_labels()
            for tool_label in tool_labels:
                if self.has_tool(tool_label):
                    self.add_renderers_to_tool(tool_label, renderers)
                else:
                    tool = element.tool_mapping[tool_label](renderers=renderers)
                    self.add_tool(tool_label, tool)
            element.add_autofill()
        return element

    def add_lines(self, *args, allow_edit=False, **kwargs) -> Lines:

    def add_free_point(self, *args, allow_edit=True, **kwargs):
        assert allow_edit, 'Edit-only tool'
        assert 'pointset' in kwargs
        free_draw = FreePointAdd(*args, fig=self, editable=allow_edit, **kwargs)

        renderers = self.get_renderers(free_draw, filter_glyphs=free_draw.editable_glyphs())
        tool_label = free_draw.tool_label()
        if self.has_tool(tool_label):
            raise RuntimeError('Only one free point add currently supported')
        else:
            tool = free_draw.make_freeadd_tool(renderers=renderers)
            self.add_tool(tool_label, tool)
        free_draw.add_autofill()
        return free_draw

[docs]    def add_edit_span(self, *args, allow_edit=True, **kwargs) -> DualEditSpan:
        """
        Add an :class:`.DualEditSpan` object to the figure

        Allows interactively selecting a 1D range along an axis with visual feedback
        """
        assert allow_edit
        span = DualEditSpan(*args, fig=self, **kwargs)
        tool_label = span.tool_name()
        if self.has_tool(tool_label):
            raise RuntimeError('Only one editable span allowed per dimension!')
        tool = span.make_span_tool()
        self.add_tool(tool_label, tool)
        span.setup_callback()
        return span
