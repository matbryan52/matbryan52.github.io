[docs]class Cursor(PointSet):
    """
    Adds an interactive point cursor to the figure

    Used to provide interaction where a user specifies an :code:`(x, y)` position
    in the figure primary axes.

    Parameters
    ----------
    marker : str, optional
        Bokeh marker specification string to use for the cursor,
        by default 'circle_dot'
    pos_xy : tuple[float, float], optional
        The initial positions of the cursor point, by default None,
        which will not create a point. The user can create the point
        by double clicking the figure with the tool selected.
    image : BokehImage, optional
        A BokehImage instance used to define the initial point, which will
        be placed in the centre of the image. If :code:`pos_xy` is provided,
        the tuple values take precendence.
    **kwargs : dict, optional
        Optional :code:`kwargs` passed to the point Glyph used for the cursor
    """
    @staticmethod
    def make_cursor_tool(renderers=None):
        if renderers is None:
            renderers = []
        return PointDrawTool(name='Cursor',
                             description='Draw cursor points on figure',
                             renderers=renderers,
                             add=True,
                             drag=True,
                             num_objects=1,
                             empty_value=PointSet._default_fill_value,
                             custom_icon=icons.cursor_icon())

    @staticmethod
    def tool_label():
        return 'cursor'

    def build_figure(self, marker='circle_dot', pos_xy: tuple[float, float] = None,
                     image: BokehImage = None, **kwargs):
        self.cds = ColumnDataSource({'cx': [], 'cy': []})
        if pos_xy is None and image is not None:
            pos_xy = image.current_centre_xy()
        if pos_xy is not None:
            self.cds.data.update({'cx': [pos_xy[0]], 'cy': [pos_xy[1]]})
        self._cursor_glyph = Scatter(marker=marker,
                                     x='cx',
                                     y='cy',
                                     line_color=kwargs.pop('line_color', 'red'),
                                     line_width=kwargs.pop('line_width', 3),
                                     fill_alpha=kwargs.pop('fill_alpha', 0.),
                                     size=kwargs.pop('size', 10),
                                     **kwargs)
        self.add_glyph(self.cds, self._cursor_glyph)

    @property
    def cursor_glyph(self):
        try:
            return self._cursor_glyph
        except AttributeError:
            return None


[docs]    def get_posxy(self, data: dict = None) -> Union[tuple[float, float],
                                                    tuple[None, None]]:
        """
        Return the current :code:`(x, y)` floating point position of the cursor,
        if defined. Returns the position in the primary axes coordinates.

        Parameters
        ----------
        cds : ColumnDataSource, optional
            If supplied, use this data source for returning the position,
            else use the internal data source of the cursor.

        Returns
        -------
        Union[tuple[float, float], tuple[None, None]]
            Position of the cursor :code:`(x, y)`, or :code:`(None, None)` if invalid
        """
        if data is None:
            data = self.cds.data
        xvals = data['cx']
        yvals = data['cy']
        if xvals and yvals:
            xv = xvals[0]
            yv = yvals[0]
            return xv, yv
        return None, None



[docs]    def get_int_posxy(self, data=None,
                      bounds: tuple[int, int, int, int] = None) -> Union[tuple[int, int],
                                                                         tuple[None, None]]:
        """
        Return the current :code:`(x, y)` integer point position of the cursor,
        if defined. Returns the position in the primary axes coordinates, rounded
        to the nearest integer position.

        If bounds are supplied, the point is only considered valid if within these
        integer bounds.

        Parameters
        ----------
        cds : ColumnDataSource, optional
            If supplied, use this data source for returning the position,
            else use the internal data source of the cursor.
        bounds : tuple[int, int, int, int], optional
            Valid bounds for the position, :code:`left, top, right, bottom`

        Returns
        -------
        Union[tuple[int, int], tuple[None, None]]
            Position of the cursor :code:`(x, y)`, or :code:`(None, None)` if invalid
        """
        xv, yv = self.get_posxy(data=data)
        if not (xv and yv):
            return None, None
        xv = int(np.round(xv))
        yv = int(np.round(yv))
        if bounds:
            assert all(isinstance(e, int) for e in bounds)
            left, top, right, bottom = bounds
            if not (left <= xv < right):
                xv = None
            if not (top <= yv < bottom):
                yv = None
        return xv, yv


    def get_int_posyx(self, *args, **kwargs):
        return self.get_int_posxy(*args, **kwargs)[::-1]

    def move_to_posxy(self, posxy: tuple[float, float]):
        x, y = posxy
        self.cds.data.update({'cx': [x], 'cy': [y]})

    def move_to_posyx(self, posyx: tuple[float, float]):
        return self.move_to_posxy(posyx[::-1])

        disk_kwargs = kwargs.pop('disk_kwargs', {})
        super().build_figure(*args, **kwargs)
        self.add_disk(initial_radius, **disk_kwargs)

    """
    Adds an interactive point cursor which can be switched between
    point, disk and annulus representations via a selection box.

    Supports overloading of the selection options so that multiple
    point / disk / annulus cursors can be defined with different
    names displayed to the user.

    See :class:`~Cursor` for the remainder of the signature.

    Parameters
    ----------
    annotation_kwargs : dict[str, Any], optional
        Optional dictionary of :code:`kwargs` which are passed
        to the glyph rendering the Disk/Annulus, i.e. directly to
        the Bokeh Glyph, which can be used to set style options.
    """
    def __init__(self, *args, **kwargs):
        self._annotation_renderer = None
        self._name_mapping = {}
        super().__init__(*args, **kwargs)

    def build_figure(self, *args, **kwargs):
        self._annotation_kwargs = kwargs.pop('annotation_kwargs', {})
        super().build_figure(*args, **kwargs)


[docs]    def get_params(self) -> dict[str, Union[str, float]]:
        """
        Get the current cursor parameters

        Returns
        -------
        dict[str, Union[str, float]]
            Dictionary containing the currently selected cursor :code`mode`
            (displayed name) as well as the point position (:code:`'cx'`,
            :code:`'cy'`), and any parameters defined on the cursor
            (:code:`'r0'`, :code:`'r1'` as appropriate).
        """
        name = self.selected_name
        cx, cy = self.get_posxy()
        params = {'mode': name, 'cx': cx, 'cy': cy}
        params.update(**self._param_getter[self.selected_mode]())
        return params


    @property
    def selected_name(self):
        return self.cursor_select.value

    @property
    def selected_mode(self):
        return self._name_mapping[self.cursor_select.value]

    @property
    def cursor_mode(self):
        try:
            return self.cds.data['cursor_mode'][0]
        except KeyError:
            None

    @property
    def cursor_select(self):
        try:
            return self._cursor_select
        except AttributeError:
            return self.make_cursor_select()


[docs]    def make_cursor_select(self,
                           title: str = 'Cursor type',
                           mapping: dict[str, Union[Literal['point'],
                                                    Literal['disk'],
                                                    Literal['annulus']]] = None,
                           value: str = None,
                           **kwargs) -> Select:
        """
        Create the selection box used for switching the cursor mode. Supports
        the :code:`mapping` argument which is used to overload the cursor with
        extra modes with custom display names.

        Automatically sets up callbacks so that the cursor is responsive.

        Parameters
        ----------
        title : str, optional
            Label to apply to the selection box, by default 'Cursor type'
        mapping : dict[str, Union[Literal['point'], Literal['disk'], Literal['annulus']]], optional
            Dictionary mapping from a displayed name for a cursor and
            one of :code:`'point'`, :code:`'disk'`, :code:`'annulus'`,
            by default None in which case the displayed names are these
            strings but capitalized.
        value : str, optional
            The inital state of the cursor, by default None in which case
            the state is the first of the options.
        **kwargs : dict, optional
            Optional :code:`kwargs` passed to the Select constructor, can be
            used for to set style options.
        Returns
        -------
        Select
            The selection box for the layout
        """
        if mapping is None:
            options = [o.capitalize() for o in self._annotation_constructors.keys()]
            self._name_mapping = {o: o.lower() for o in options}
        else:
            assert isinstance(mapping, dict)
            assert mapping
            assert set(mapping.values()) == set(self._annotation_constructors.keys())
            options = [*mapping.keys()]
            self._name_mapping = mapping

        if value is None:
            value = options[0]
        else:
            assert value in options

        self._cursor_select = Select(title=title,
                                     options=options,
                                     value=value,
                                     **kwargs)
        self.cds.data.update({'cursor_mode': [self._name_mapping[value]]})
        self.cds.data.update({'r0': [5.], 'r1': [10.]})
        self.cursor_select.on_change('value', self.set_annotation)
        self.cursor_select.trigger('value', None, value)
        return self.cursor_select


    @property
    def _annotation_constructors(self):
        return {'point': self._add_point,
                'disk': self._add_disk,
                'annulus': self._add_annulus}

    @property
    def _slider_setup(self):
        return {'point': self._set_slider_point,
                'disk': self._set_slider_disk,
                'annulus': self._set_slider_annulus}

    @property
    def _param_getter(self):
        return {'point': self._params_point,
                'disk': self._params_disk,
                'annulus': self._params_annulus}

    def set_annotation(self, attr, old, new):
        assert attr == 'value'
        if new == old:
            return

        if self._annotation_renderer is not None:
            self.fig.remove_renderer(self._annotation_renderer)
        self._annotation_renderer = None
        self._annotation_constructors[self.selected_mode]()

    def _add_point(self):
        self.cds.data.update({'cursor_mode': ['point']})
        self._set_slider_point()
        assert self._annotation_renderer is None

    def _set_slider_point(self, *args):
        if self.radius_slider is not None:
            self.radius_slider.disabled = True

    def _params_point(self):
        return {}

    def _add_disk(self):
        self.cds.data.update({'cursor_mode': ['disk']})
        radius = self.cds.data.get('r1', [100.])[0]
        self._set_slider_disk(radius)
        self._annotation_renderer = self.add_disk(radius, **self._annotation_kwargs)

    def _set_slider_disk(self, r0, *args):
        if self.radius_slider is not None:
            self.radius_slider.disabled = False
            self.radius_slider.value = (0., r0)

    def _params_disk(self):
        return {'r0': self.cds.data['r0'][0]}

    def _add_annulus(self):
        self.cds.data.update({'cursor_mode': ['annulus']})
        r0 = self.cds.data.get('r0', [100.])[0]
        r1 = self.cds.data.get('r1', [350.])[0]
        if (r0 == 0) or (r0 == r1):
            r0 = 0.33 * r1
        self._set_slider_annulus(r0, r1)
        self._annotation_renderer = self.add_annulus(r0, r1, **self._annotation_kwargs)

    def _set_slider_annulus(self, r0, r1):
        if self.radius_slider is not None:
            self.radius_slider.disabled = False
            self.radius_slider.value = (r0, r1)

    def _params_annulus(self):
        return {'r0': self.cds.data['r0'][0],
                'r1': self.cds.data['r1'][0]}

    @property
    def radius_slider(self):
        try:
            return self._slider
        except AttributeError:
            return None


[docs]    def make_radius_slider(self, *, r0: float, r1: float, max_r: float,
                           min_r: float = 0.01, step: float = 0.01,
                           name: str = 'Radius', **kwargs) -> RangeSlider:
        """
        Create the radius slider used to modify the cursor size. Uses a
        two-point RangeSlider to set a lower/upper limit. When in disk or
        point mode the limit are appropriate ignored when updating.

        Automatically sets up callbacks so that the cursor is responsive.

        Parameters
        ----------
        r0 : float
            Inner radius
        r1 : float
            Outer radius
        max_r : float
            Maximum radius
        min_r : float, optional
            Minimum radius, by default 0.01
        step : float, optional
            Radius step size, by default 0.01
        name : str, optional
            Label on the slider, by default 'Radius'
        **kwargs : dict, optional
            Optional :code:`kwargs` passed to the RangeSlider constructor, can be
            used for to set style options.

        Returns
        -------
        RangeSlider
            The radius range slider for the layout
        """
        self._slider = AnnulusSliderMixin._make_slider(r0, r1, max_r, min_r=min_r,
                                                       step=step, name=name, **kwargs)
        self._slider.jscallback(value=self._radius_js(), args={'cds': self.cds})
        cursor_mode = self.cursor_mode
        if cursor_mode is not None:
            self._slider_setup[cursor_mode](r0, r1)
        self.cds.data.update({'r0': [r0], 'r1': [r1]})
        return self.radius_slider

    cds.data.r0.fill(cb_obj.value[0]);
    cds.data.r1.fill(cb_obj.value[1]);
}
cds.change.emit();
// see https://discourse.bokeh.org/t/synchronizing-columndatasource
// -with-bokeh-server-with-when-editing-data-in-js-callbacks/4355/4
cds.data = cds.data;
cds.properties.data.change.emit();
'''
