import param
import panel as pn
from bokeh.models import Column as BkColumn, Row as BkRow



[docs]class AutoLayout(pn.layout.ListPanel):
    """
    A ListLike layout which can be switched from
    row-like to column-like as appropriate to the contents

    As the layout is list-like it supports all the usual methods
    like append, extend etc.

    By default acts as a Column to avoid error cases
    """
    lo = param.String('Any')
    sizing_mode = 'stretch_width'
    margin = (0, 0)

    _bokeh_model = BkColumn
    _manual_params = ['lo']
    __abstract = False
