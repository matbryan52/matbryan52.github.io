[docs]class PanelInterface(param.Parameterized):
    """
    Mixin handling the Panel application aspects of the UI
    in particular the creation and management of :code:`Window`
    objects.

    Constructs the main application using a Panel template
    and is responsible for switching the contents of
    main / sidebar / modal when Windows are loaded/unloaded.
    """

    error_trigger = param.Event()

    def __init__(self, panel_spec: Optional[dict] = None, **params):
        super().__init__(**params)

        self._panel_spec = panel_spec
        if self._panel_spec is None:
            self._panel_spec = {}

        self._panel = None
        self._current_window = None
        self._home_window_name = 'home'
        self._windows = {}
        self._options_callbacks = []

        self.setup_triggers()

    @property
    def session_id(self):
        try:
            # Unique bokeh session id linked to this Session to
            # make the user session re-connectable
            # This will not work well in a multi-user single-server
            # context but for now that's not the goal, this can
            # be later solved with proper auth hooks
            return f'?bokeh-session-id={self._session_id}'
        except AttributeError:
            return ''

    @session_id.setter
    def session_id(self, val: str):
        self._session_id = val

    @property
    def html_endpoint(self):
        try:
            # The endpoint the site is being served on
            # Used to set a URL in the titlebar
            return self._html_endpoint
        except AttributeError:
            return '/'

    @html_endpoint.setter
    def html_endpoint(self, val: str):
        if not val.startswith('/'):
            val = '/' + val
        self._html_endpoint = val

    @property
    def site_url_id(self):
        return self.html_endpoint + self.session_id

    def windows(self):
        return self._windows

    @property
    def panel(self) -> pn.template.base.Template:
        if not self._has_panel:
            self._build_interface()
        return self._panel

    @property
    def _has_panel(self) -> bool:
        return self._panel is not None

    @property
    def _template_class(self) -> Type[pn.template.base.Template]:
        return self._panel_spec.pop('template', pn.template.FastListTemplate)

    @property
    def _title(self) -> str:
        return self._panel_spec.pop('title', 'LiberTEM')

    @property
    def _logo(self) -> str:
        return self._panel_spec.pop('logo', 'app/static/libertem_logo.png')

    @property
    def _favicon(self) -> str:
        return self._panel_spec.pop('favicon', 'app/static/favicon.ico')

    @property
    def _bkg_color(self) -> str:
        return self._panel_spec.pop('bkg_color', '#aaaaaa')

    def _build_interface(self):
        assert self._panel is None
        self._panel = self._template_class(title=self._title,
                                           logo=self._logo,
                                           site_url=self.site_url_id,
                                           header_background=self._bkg_color,
                                           favicon=self._favicon,
                                           main_layout='',
                                           **self._panel_spec)
        self._panel.sidebar.append(pn.Column(margin=(0, 0)))
        self._panel.main.append(pn.Column(margin=(0, 0)))
        self._panel.modal.append(pn.Column(margin=(0, 0)))
        self._panel.header.append(pn.Row())

        self.header.append(pn.Spacer(width=1, height=1, sizing_mode='fixed', align='center'))
        self.header.append(pn.pane.HTML(object=self._format_subtitle(''),
                                        width=200,
                                        height=50,
                                        sizing_mode='fixed',
                                        align='center'))
        self.header.append(pn.Spacer(height=1,
                                     sizing_mode='stretch_width',
                                     align='center'))
        self._error_bar = pn.pane.HTML(object=self._format_central_text('', 'left'),
                                       height=50,
                                       width=300,
                                       sizing_mode='fixed',
                                       visible=True,
                                       align='center')
        self.header.append(self._error_bar)
        self.header.append(pn.pane.HTML(object=self._format_central_text('Open windows:', 'right'),
                                        width=120,
                                        height=50,
                                        sizing_mode='fixed'))
        self._window_selector = pn.widgets.Select(value='',
                                                  options=[],
                                                  max_width=300,
                                                  height=50,
                                                  sizing_mode='stretch_width')

        def _update_window_select(options):
            self._window_selector.options = options
        self.register_window_cb(_update_window_select)

        self.header.append(self._window_selector)
        go_button = pn.widgets.Button(name='Switch',
                                      button_type='default',
                                      width=75,
                                      sizing_mode='stretch_height',
                                      margin=(7, 5, 5, 5))
        self.header.append(go_button)

        def _switch_window(*events):
            select_value = self._window_selector.value
            if select_value != self._current_window:
                self.switch_to_window(select_value)

        go_button.on_click(_switch_window)

        def _go_home(*events):
            if self._current_window != self._home_window_name:
                self.switch_to_window(self._home_window_name)

        home_button = pn.widgets.Button(name='Home',
                                        button_type='default',
                                        width=75,
                                        # height=35,
                                        sizing_mode='stretch_height',
                                        margin=(7, 5, 5, 5))
        self.header.append(home_button)

        home_button.on_click(_go_home)

        self.entrypoint()

    def set_subtitle(self, v):
        self.header[1].object = self._format_subtitle(v)

    @staticmethod
    def _format_subtitle(v):
        return f"""
<div class="title">
    <span style="text-align: left; vertical-align: middle;">{v}</span>
</div>"""

    @staticmethod
    def _format_central_text(v, align, height=50):
        return (f'<div style="line-height: {height}px; '
                f'text-align: {align}; vertical-align: middle;">{v}</div>')

    def load_window(self, window):
        self.set_main(window.main)
        self.set_sidebar(window.sidebar)
        self.set_modal(window.modal)
        self.set_subtitle(window.subtitle)

    def add_window(self, window, name=None, suffix=None, switch=False):
        if not window._creation_time:
            window.create()
        assert window._creation_time is not None, ('Must call super() on '
                                                   f'{window.__class__}.create()')
        if name is None:
            name = window.shortname
        if suffix is None:
            suffix = random_hash()
        if suffix is False:
            fullname = f'{name}'
        else:
            fullname = f'{window.shortname}-{suffix}'
        if window._selectable:
            fullname = self.add_window_classname(fullname, window=window)
        if fullname in self._windows.keys():
            return self.add_window(window, name=name, suffix=None, switch=switch)

        self._windows[fullname] = window
        self.update_window_lists()

        if window._selectable:
            self.set_selected_window(fullname)
            if switch:
                self.switch_to_window(fullname)
        return fullname

    @staticmethod
    def add_window_classname(name, window=None, ident=None):
        if ident is not None:
            pass
        elif window is not None:
            ident = window.subtitle
        else:
            raise
        return f'{name} [{ident}]'

    @staticmethod
    def remove_window_classname(name):
        try:
            start = name.index('[')
            stop = name.index(']')
            assert stop + 1 == len(name)
        except (ValueError, AssertionError):
            return name
        return name[:start].strip()

    def remove_window(self, window_name):
        if self.selected_window() == window_name:
            self.set_selected_window()
        window = self._windows.pop(window_name, None)
        if window:
            self.update_window_lists()
            del window

    def get_window(self, window_name):
        return self._windows.get(window_name, None)

    def get_window_name(self, window, with_classname=True):
        for k, v in self.windows().items():
            if v is window:
                if not with_classname:
                    return self.remove_window_classname(k)
                return k
        raise RuntimeError(f'Window not known to Session {window}')

    def rename_window(self, window_name, new_name):
        # Could add a modal popup for an invalid name here
        if window_name not in self.all_window_names() or not new_name:
            # No name supplied or window doesn't exist
            return False

        selected = window_name == self.selected_window()
        new_name = str(new_name)

        window = self._windows.get(window_name, None)
        if not window._selectable:
            # Can't rename a non-selectable window
            return False
        new_name = self.add_window_classname(new_name, window)

        if new_name in self.all_window_names():
            # Name already exists
            return False

        self._windows[new_name] = self._windows.pop(window_name)
        self.update_window_lists()
        if selected:
            self.set_selected_window(window_name=new_name)
        return True

    def switch_to_window(self, window_name):
        if not window_name or not len(self._windows.keys()):
            return
        if self._current_window == window_name:
            return
        assert window_name in self._windows.keys(), f'Unrecognized window name: {window_name}'
        window = self._windows[window_name]
        self._current_window = window_name
        self.load_window(window)
        window.post_switch_hook()

    def selectable_window_names(self):
        selectable = {v._creation_time: k for k, v in self.windows().items() if v._selectable}
        return [selectable[t] for t in sorted(selectable)]

    def all_window_names(self):
        return [k for k in self.windows().keys()]

    def update_window_lists(self):
        options = self.selectable_window_names()
        for cb in self._options_callbacks:
            cb(options)

    def selected_window(self):
        return self._window_selector.value

    def set_selected_window(self, window_name=None):
        available_names = self.selectable_window_names()
        if window_name is None:
            self._window_selector.value = available_names[0]
        elif window_name in available_names:
            self._window_selector.value = window_name
        else:
            raise ValueError(f'Unrecognized window name {window_name}')

    def register_window_cb(self, cb):
        self._options_callbacks.append(cb)

    @property
    def sidebar(self):
        return self.panel.sidebar[0]

    def set_sidebar(self, layout, clear=True):
        if clear:
            self.sidebar.clear()
        self.sidebar.append(layout)

    @property
    def main(self):
        return self.panel.main[0]

    def set_main(self, layout, clear=True):
        if clear:
            self.main.clear()
        self.main.append(layout)

    @property
    def header(self):
        return self.panel.header[0]

    @property
    def modal(self):
        return self.panel.modal[0]

    def set_modal(self, layout, clear=True):
        if clear:
            self.modal.clear()
        self.modal.append(layout)

    def open_modal(self):
        self.panel.open_modal()

    def close_modal(self):
        self.panel.close_modal()

    def clear_current_layout(self):
        self.main.clear()
        self.modal.clear()
        self.sidebar.clear()

    def entrypoint(self):
        from .applications.home import HomePage
        home = HomePage(self)
        name = self.add_window(home, name=self._home_window_name, suffix=False)
        assert name == self._home_window_name
        self.switch_to_window(self._home_window_name)

    def setup_triggers(self):
        self._error_message = ''
        self._error_timeout = 2
        self.param.watch(self._display_error_async, 'error_trigger')

    def display_error(self, message):
        self._error_message = message
        self.param.trigger('error_trigger')

    async def _display_error_async(self, *values):
        self._error_bar.object = self._format_central_text(self._error_message, 'left')
        await asyncio.sleep(float(self._error_timeout))
        self._error_bar.object = self._format_central_text('', 'left')




[docs]class Session(PanelInterface):
    """
    Main entry-point which handles the functionality of the application,
    in particular managing backends, datasets, results and the tools
    which can be launched within the UI.
    """

    _application_classes = available_applications()

    def __init__(self, panel_spec=None):
        super().__init__(panel_spec=panel_spec)

        self._backend_definitions = backend_defs
        self._open_backends = {}
        self._datasets = {}
        self._datasets_cbs = []
        self._paramsets = []
        self._results = []

    def tool_mapping(self):
        return {t.subtitle: t for t in self._application_classes}

    def launch_tool(self, tool_name, suffix):
        if suffix is not None:
            suffix = str(suffix)
        if not suffix:
            suffix = None
        if isinstance(tool_name, str):
            tool_cls = self.tool_mapping()[tool_name]
        else:
            assert issubclass(tool_name, Window)
            tool_cls = tool_name
        if tool_cls._needs_backend and tool_cls._needs_backend not in self.active_backends:
            return
        tool = tool_cls(self)
        name = self.add_window(tool, suffix=suffix, switch=True)
        return name

    def register_datasets_cb(self, cb):
        self._datasets_cbs.append(cb)

    def rename_dataset(self, dataset_name, new_name):
        if dataset_name not in self.open_datasets():
            return
        if not new_name or not isinstance(new_name, str):
            return

        ds = self._datasets.get(dataset_name)
        new_fullname = self.add_window_classname(new_name, ident=ds.ident)
        if new_fullname not in self._datasets:
            self._datasets[new_fullname] = self._datasets.pop(dataset_name)
            self.update_dataset_lists()
            # Try to rename any windows which reference this dataset
            to_update = []
            for window in self.windows().values():
                if hasattr(window, 'datasets'):
                    if ds in window.datasets:
                        to_update.append(window)
            for window in to_update:
                try:
                    window.update_name()
                except AttributeError:
                    pass

    def remove_dataset(self, dataset_name):
        if dataset_name not in self.open_datasets():
            return

        ds = self._datasets.pop(dataset_name)
        self.update_dataset_lists()
        # Try to unload any windows which use this dataset
        to_unload = []
        for window in self.windows().values():
            if hasattr(window, 'datasets'):
                if ds in window.datasets:
                    to_unload.append(window)
        for window in to_unload:
            try:
                window.unload()
            except AttributeError:
                pass
        del ds

    def get_dataset(self, dataset_name):
        return self._datasets.get(dataset_name, None)

    def get_dataset_name(self, dataset, with_classname=True):
        for k, v in self.open_datasets().items():
            if v is dataset:
                if not with_classname:
                    return self.remove_window_classname(k)
                return k
        raise RuntimeError('Dataset not know to Session')

    def open_datasets(self):
        return self._datasets

    def ordered_dataset_names(self):
        opened = {v._creation_time: k for k, v in self.open_datasets().items()}
        return [opened[t] for t in sorted(opened)]

    def update_dataset_lists(self):
        options = self.ordered_dataset_names()
        for cb in self._datasets_cbs:
            cb(options)

    def get_dataset_selector(self, display_name='Dataset:', **kwargs):
        selector = pn.widgets.Select(name=display_name,
                                     options=self.ordered_dataset_names(),
                                     value='')

        def _update_options(*events):
            selector.options = self.ordered_dataset_names()

        # Need a mechanism to remove cb and selector
        self.register_datasets_cb(_update_options)
        return selector

    def add_dataset(self, config_window, params):
        if isinstance(config_window, Window):
            config_window = type(config_window)
        backend = self.get_backend(config_window._needs_backend)
        dataset = backend.load_dataset(params, config_window)
        if not dataset:
            return False

        shortname = config_window.shortname
        suffix = random_hash()
        fullname = f'{shortname}-{suffix}'
        fullname = self.add_window_classname(fullname, window=config_window)

        self._datasets[fullname] = dataset
        self.update_dataset_lists()
        return fullname

    @property
    def possible_backends(self):
        return self._backend_definitions

    @property
    def open_backends(self):
        return self._open_backends

    @property
    def active_backends(self):
        return {k: v.is_active for k, v in self.open_backends.items()}

    @property
    def lt(self) -> 'LiberTEMMixin':
        return self.get_backend('lt')

    def get_backend(self, backend_name):
        backend = self.open_backends.get(backend_name, None)
        if backend is None:
            backend_def = self.possible_backends.get(backend_name, None)
            assert backend_def is not None, f'Unrecognized backend name {backend_name}'
            self.open_backends[backend_name] = backend_def.constructor(self)
            backend = self.get_backend(backend_name)
            backend.post_init_tasks()
        return backend
