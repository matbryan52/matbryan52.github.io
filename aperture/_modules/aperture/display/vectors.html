[docs]class VectorsOverlay(GlyphBase):
    """
    Adds two lines to the figure at right-angles, with options
    to control color and add text labels. Also has functionality
    to interactively set the vector angle using a slider.

    If initialized on a :class:`.PointSet`,
    vector origin location will update based on either changes to the
    :class:`PointSet` data source, or to the value of the selection box
    returned by :meth:`~VectorsOverlay.point_select`.

    This class currently requires the user to call
    :meth:`~VectorsOverlay.setup_callbacks` to configure which
    types of interactivity should be created. This should be the subject
    of a future refactoring as needed.

    Parameters
    ----------
    pointset : PointSet, optional
        A :class:`~PointSet` to use as the source for the vector origin,
        by default None in which case :code:`pos_xy` must be supplied.
    pos_xy : tuple[float, float], optional
        Values to use for the vector origin, by default None in which
        case :code:`pointset` must be supplied.
    length : float, optional
        The vector length in data units, by default 100.
    colors : tuple[str, str], optional
        A 2-tuple of color strings, one for each vector, by default None
        in which case colors are red and blue.
    labels : tuple[str, str], optional
        2-tuple of label strings to applied to the end of each vector,
        by default None in which case there are no labels applied.
    text_length_mult : float, optional
        A multiplier applied to length to defined text label origin,
        by default 1.1
    initial_idx : int, optional
        The initial index in the :code:`pointset` datasource to which to
        attach the vector origin, by default 0
    initial_angle : float, optional
        The inital angle of the vectors in radians, relative to a horizontal-right
        origin, by default 0.
    labels_kwargs : dict, optional
        Optional :code:`kwargs` passed to the :class:`~TextOverlay` constructor,
        which can be used to set style options on the labels
    **kwargs : dict, optional
        Optional :code:`kwargs` passed to the Bokeh MultiLine constructor, which
        can be used to set style options on the vector glyph.
    """
    @property
    def rotation_slider(self) -> Slider:
        """
        Get or make the rotation slider to control the vectors

        If control is desired over the slider params, call
        :meth:`~VectorsOverlay.make_rotation_slider` first.

        Returns
        -------
        Slider
            The rotation slider
        """
        try:
            return self._rotation_slider
        except AttributeError:
            return self.make_rotation_slider()


[docs]    def make_rotation_slider(self, start_angle: float = None,
                             name: str = 'Angle (rad)', **kwargs) -> Slider:
        """
        Make the rotation slider to control the vectors, in radians


        Parameters
        ----------
        start_angle : float, optional
            Set the initial slider angle, by default 0. radians
        name : str, optional
            The label to apply to the slider, by default 'Angle'
        **kwargs : dict, optional
            Optional :code:`kwargs` passed to the Slider constructor,
            which can be used to set style options.

        Returns
        -------
        Slider
            The rotation slider
        """
        if start_angle is None:
            start_angle = 0.
            if self.cds is not None:
                start_angle = self.cds.data['angle'][0]
        assert -np.pi < start_angle <= np.pi
        self._rotation_slider = Slider(title=name,
                                       value=start_angle,
                                       start=-np.pi,
                                       end=np.pi,
                                       step=0.01,
                                       **kwargs)
        return self.rotation_slider


    def build_figure(self,
                     pointset: PointSet = None,
                     pos_xy: tuple[float, float] = None,
                     length: float = 100.,
                     colors: tuple[str, str] = None,
                     labels: tuple[str, str] = None,
                     text_length_mult: float = 1.1,
                     initial_idx: int = 0,
                     initial_angle: float = 0.,
                     labels_kwargs: dict = None,
                     **kwargs):
        if colors is None:
            colors = ['red', 'blue']
        assert len(colors) == 2

        if pointset is not None:
            assert pos_xy is None, 'can build VectorsOverlay either from PointSet or fixed position'
            pos_xy = self._get_initial_pos_xy(pointset, initial_idx)
        else:
            assert pos_xy is not None, ('can build VectorsOverlay either from '
                                        'PointSet or fixed position')

        # Build vectors
        self.cds = ColumnDataSource({**self._vectors_init(pos_xy, initial_angle, length),
                                     'line_color': colors,
                                     'length': [length] * 2,
                                     'angle': [initial_angle] * 2,
                                     'idx': [initial_idx] * 2})
        glyph = MultiLine(xs="xs",
                          ys="ys",
                          line_color='line_color',
                          line_width=kwargs.pop('line_width', 2),
                          **kwargs)
        self.add_glyph(self.cds, glyph)

        # Add Text overlay for labels
        self._text = None
        if labels is not None:
            if labels_kwargs is None:
                labels_kwargs = {}
            assert len(labels) == 2
            assert all(isinstance(label, str) for label in labels)
            initial_pos_text = self._vectors_init(pos_xy, initial_angle, length,
                                                  mult=text_length_mult)
            initial_pos_text = [(initial_pos_text['xs'][0][1], initial_pos_text['ys'][0][1]),
                                (initial_pos_text['xs'][1][1], initial_pos_text['ys'][1][1])]
            with self.fig.as_parent(self):
                self._text = self.fig.add_text_overlay(pos_xy=initial_pos_text,
                                                    strings=labels,
                                                    other_data={'text_color': colors,
                                                                'length_mult': [text_length_mult,
                                                                                text_length_mult]},
                                                    text_color='text_color',
                                                    text_baseline=labels_kwargs.pop('text_baseline',
                                                                                    'middle'),
                                                    text_align=labels_kwargs.pop('text_align',
                                                                                 'center'),
                                                    **labels_kwargs)

    @staticmethod
    def _get_initial_pos_xy(pointset, initial_idx):
        try:
            return pointset.cds.data['cx'][initial_idx], pointset.cds.data['cy'][initial_idx]
        except IndexError:
            raise IndexError(f'Initialising VectorsOverlay with initial_idx {initial_idx}'
                             f'but PointSet has length {len(pointset.cds.data["cx"])}')

    @staticmethod
    def _vectors_init(pos_xy, angle, length, mult=1.):
        cx, cy = pos_xy
        dx0, dy0 = np.cos(angle) * length * mult, np.sin(angle) * length * mult
        dx1, dy1 = np.cos(angle + np.pi/2) * length * mult, np.sin(angle + np.pi/2) * length * mult
        return {'xs': [[cx, cx + dx0], [cx, cx + dx1]],
                'ys': [[cy, cy + dy0], [cy, cy + dy1]]}


[docs]    def setup_callbacks(self,
                        rotation: bool = True,
                        pointset: PointSet | None = None,
                        selectable: bool = False):
        """
        Configure callbacks on the VectorsOverlay depending on requirements

        Currently the user *must* call this to set up the desired callbacks.

        Parameters
        ----------
        rotation : bool, optional
            Add rotation via the slider returned by :meth:`~VectorsOverlay.rotation_slider`,
            by default True. If control is desired over the slider creation,
            call :meth:`VectorsOverlay.self.make_rotation_slider` first
            to set this property.
        pointset : Optional[PointSet], optional
            Link the vector origin to the supplied :class:`~PointSet`,
            by default None in which case the origin is fixed.
        selectable : bool, optional
            If linking to a :class:`~PointSet` enable selection of the
            point to follow by setting selectable to True, by default False.
            If set up, use the select box returned by :meth:`~VectorsOverlay.point_select`
            to control the vector location.
        """
        args = {'vector_source': self.cds}

        if self._text is not None:
            args['text_source'] = self._text.cds
            text_code = self._text_cb_js()
        else:
            text_code = ''

        if pointset is not None:
            args['point_source'] = pointset.cds
            if selectable:
                if self.point_select is None:
                    self.get_point_select(pointset)
                args['point_select'] = self.point_select

        if rotation:
            args['angle_slider'] = self.rotation_slider
            rotation_code = self._vectors_cb_rotation_js() + self._vectors_cb_base_js() + text_code
            callback_rot = CustomJS(args=args,
                                    code=rotation_code)
            self.rotation_slider.js_on_change('value', callback_rot)

        if pointset is not None:
            pointset_code = self._vectors_cb_pointset_js() + self._vectors_cb_base_js() + text_code
            callback_pointset = CustomJS(args=args,
                                         code=pointset_code)
            pointset.cds.js_on_change('data', callback_pointset)

            if selectable:
                callback_select = CustomJS(args=args,
                                           code=pointset_code)
                self.point_select.js_on_change('value', callback_select)



[docs]    def move_centre(self, pos_xy: tuple[float, float]):
        """
        Moves the centre of the vectors display to pos_xy

        Parameters
        ----------
        pos_xy : tuple[float, float]
            The new coordinates of the vectors origin
        """
        current_angle = self.rotation_slider.value
        original_length = self.cds.data['length'][0]
        new_cds_coords = self._vectors_init(pos_xy, current_angle, original_length)
        self._update_cds_datadict(self.cds, new_cds_coords)
        if self._text is not None:
            text_mult = self._text.cds.data['length_mult'][0]
            new_text_cds_coords = self._vectors_init(pos_xy,
                                                     current_angle,
                                                     original_length,
                                                     mult=text_mult)
            self._text._update_cds_datadict(self._text.cds, new_text_cds_coords)

    @staticmethod
    def _vectors_cb_pointset_js():
        return '''
var point_idx;
if (typeof point_select !== 'undefined') {
  if (point_source.data.hasOwnProperty('label')) {
      point_idx = point_source.data.label.findIndex(element =>  element === point_select.value)
      if (point_idx == -1) {
          point_idx = 0
      }
  } else {
      point_idx = point_select.value;
  }
} else {
  point_idx = vector_source.data.idx[0];
}
const centre_x = point_source.data.cx[point_idx]
const centre_y = point_source.data.cy[point_idx]
'''

    @staticmethod
    def _vectors_cb_rotation_js():
        return '''
const centre_x = vector_source.data.xs[0][0]
const centre_y = vector_source.data.ys[0][0]
'''

    @staticmethod
    def _vectors_cb_base_js():
        return '''
var angle;
if (typeof angle_slider !== 'undefined') {
  angle = angle_slider.value;
} else {
  angle = vector_source.data.angle[0];
}

const length = vector_source.data.length

const cos_dir = Math.cos(angle)
const sin_dir = Math.sin(angle)
const cos_dir_pi2 = Math.cos(angle + Math.PI/2)
const sin_dir_pi2 = Math.sin(angle + Math.PI/2)

vector_source.data.xs = []
vector_source.data.ys = []
vector_source.data.angle = [angle, angle]
vector_source.data.xs.push([centre_x, centre_x + length[0] * cos_dir])
vector_source.data.ys.push([centre_y, centre_y + length[0] * sin_dir])
vector_source.data.xs.push([centre_x, centre_x + length[1] * cos_dir_pi2])
vector_source.data.ys.push([centre_y, centre_y + length[1] * sin_dir_pi2])
vector_source.change.emit()
'''

    @staticmethod
    def _text_cb_js():
        return '''

text_source.data.x = []
text_source.data.y = []
text_source.data.x.push(centre_x + length[0] * text_source.data.length_mult[0] * cos_dir)
text_source.data.x.push(centre_x + length[1] * text_source.data.length_mult[1] * cos_dir_pi2)
text_source.data.y.push(centre_y + length[0] * text_source.data.length_mult[0] * sin_dir)
text_source.data.y.push(centre_y + length[1] * text_source.data.length_mult[1] * sin_dir_pi2)
text_source.change.emit()
'''




[docs]class PointSetConnector(GlyphBase):
    """
    Add a set of lines which connects points in the provided
    :class:`.PointSet`

    Connections can be integer links or pairs of strings if the pointset
    has a 'labels' column in its data source.

    The functionality surrounding labels / indexes should be refactored
    to ensure safety when handing pointsets which are being updated.

    Parameters
    ----------
    pointset : PointSet
        The pointset to link to
    connections : list[Union[tuple[int, int], tuple[str, str]]]
        Pairs of points to connect within pointset, can be int to int or
        str to str, the latter applying when the pointset is labelled
    colors : Iterable[str], optional
        Colors to use for each line, by default None in which
        case a default palette is used
    **kwargs : dict, optional
        Optional :code:`kwargs` passed to the Bokeh MultiLine constructor, which
        can be used to set style options on glyph.
    """
    def build_figure(self, *,
                     pointset: PointSet,
                     connections: list[tuple[int, int] | tuple[str, str]],
                     colors: Iterable[str] = None,
                     **kwargs):
        self._colors = colors
        self._pointset = pointset
        self._connections = ColumnDataSource({'connections': connections})
        self.cds = ColumnDataSource({**self.get_lines(self._pointset.cds.data,
                                                      self.current_connections)})

        glyph = MultiLine(xs="xs",
                          ys="ys",
                          line_color='line_color',
                          line_width=kwargs.pop('line_width', 2),
                          **kwargs)
        self.add_glyph(self.cds, glyph)

        def update_lines(attr, old, new):
            self.cds.data.update(**self.get_lines(self._pointset.cds.data,
                                                  self.current_connections))

        self._pointset.cds.on_change('data', update_lines)
        self._connections.on_change('data', update_lines)

    @property
    def current_connections(self):
        return self._connections.data['connections']

    def update_connections(self, connections):
        self._connections.data.update({'connections': connections})

    @property
    def colors(self):
        if self._colors is None:
            return bokeh_palettes.Category10_10
        else:
            return self._colors

    @staticmethod
    def _pick_point_int(data, i):
        return (data['cx'][int(i)], data['cy'][int(i)])

    @staticmethod
    def _pick_idx(data, idx):
        try:
            idx = int(idx)
        except (TypeError, ValueError):
            raise ValueError
        return idx

    @staticmethod
    def _pick_label(data, label):
        return data['label'].index(str(label))

    def get_lines(self, data: dict, connections: list[tuple]):
        lookup = self._pick_idx
        if 'label' in data.keys():
            # label column must be string
            lookup = self._pick_label

        pointlists = {'xs': [], 'ys': [], 'line_color': []}
        for (start, end), color in zip(connections, self.colors):
            # if start == end:
            #     continue
            try:
                start_idx = lookup(data, start)
                start_xy = self._pick_point_int(data, start_idx)
            except (ValueError, IndexError):
                continue
            try:
                end_idx = lookup(data, end)
                end_xy = self._pick_point_int(data, end_idx)
            except (ValueError, IndexError):
                continue
            pointlists['xs'].append([start_xy[0], end_xy[0]])
            pointlists['ys'].append([start_xy[1], end_xy[1]])
            pointlists['line_color'].append(color)
        return pointlists


[docs]    def add_veclabels(self, labels: list[str] = None, **kwargs):
        """
        Add labels to each connection between points in the centre
        of each connecting line

        Parameters
        ----------
        labels : list[str], optional
            List of labels matching list of connections, by default None
            in which case labels must already be defined on the data source
        **kwargs : dict, optional
            Optional :code:`kwargs` passed to the :class:`~TextOverlay` constructor,
            which can be used to set style options on the labels
        """
        if labels is None:
            assert 'label' in self.cds.data
        else:
            assert len(labels) == len(self.cds.data['xs'])
            self.cds.data.update({'label': labels})
        initial_pos = [(sum(x) / len(x), sum(y) / len(y)) for x, y in zip(self.cds.data['xs'],
                                                                          self.cds.data['ys'])]
        colors = self.cds.data['line_color']
        with self.fig.as_parent(self):
            self._text = self.fig.add_text_overlay(pos_xy=initial_pos,
                                                   strings=labels,
                                                   other_data={'text_color': colors},
                                                   text_color='text_color',
                                                   text_baseline=kwargs.pop('text_baseline',
                                                                            'middle'),
                                                   text_align=kwargs.pop('text_align',
                                                                         'center'),
                                                   text_font_style=kwargs.pop('text_font_style',
                                                                              'bold'),
                                                   **kwargs)

        update_labels = CustomJS(args={'lines_source': self.cds,
                                       'text_source': self._text.cds},
                                 code=self._label_update_js())
        self.cds.js_on_change('data', update_labels)


    @staticmethod
    def _label_update_js():
        return '''
text_source.data.x = lines_source.data.xs.map(xs => (xs.reduce((a, b) => a + b, 0) / xs.length));
text_source.data.y = lines_source.data.ys.map(xs => (xs.reduce((a, b) => a + b, 0) / xs.length));
text_source.data.text = lines_source.data.label;
text_source.data.text_color = lines_source.data.line_color;
text_source.change.emit();
'''




[docs]class SelectLatticeVectors(PointSetConnector):
    """
    Add `'a'` and `'b'` lattice vectors to a :code:`PointSet`, and set up callbacks
    to follow changes to the pointset data source.

    Points to use are defined by their indices.

    Creates and supplies selection boxes to change the selected points.

    Parameters
    ----------
    pointset : PointSet
        The pointset to follor.
    zero : int, optional
        The initial index of the zero position, by default 0
    a : int, optional
        The initial index of the a position, by default 1
    b : int, optional
        The initial index of the b position, by default 2
    **kwargs : dict, optional
        Optional :code:`kwargs` passed to the :class:`.PointSet`
        constructor, which can be used to set style options
    """
    def build_figure(self, *, pointset: PointSet,
                     zero: int = 0, a: int = 1, b: int = 2, **kwargs):
        self._zero_select = pointset.make_point_select(name='Zero', init_idx=zero)
        self._a_select = pointset.make_point_select(name='A', init_idx=a)
        self._b_select = pointset.make_point_select(name='B', init_idx=b)

        super().build_figure(pointset=pointset, connections=self.connections, **kwargs)
        self.add_veclabels(labels=['a', 'b'])

        self.zero_select.on_change('value', self.update_vectors)
        self.a_select.on_change('value', self.update_vectors)
        self.b_select.on_change('value', self.update_vectors)

    @property
    def connections(self):
        return [(self.zero_select.value, self.a_select.value),
                (self.zero_select.value, self.b_select.value)]

    def update_vectors(self, attr, old, new):
        self.update_connections(self.connections)

    @property
    def zero_select(self) -> Select:
        """
        Get the selection box for the zero position

        Returns
        -------
        Select
            The selection box
        """
        return self._zero_select

    @property
    def a_select(self) -> Select:
        """
        Get the selection box for the a position

        Returns
        -------
        Select
            The selection box
        """
        return self._a_select

    @property
    def b_select(self) -> Select:
        """
        Get the selection box for the b position

        Returns
        -------
        Select
            The selection box
        """
        return self._b_select
