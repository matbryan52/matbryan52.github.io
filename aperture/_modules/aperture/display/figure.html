[docs]class BokehFigureBase(ElementConstructorMixin, GeometryMixin, AxesMixin,
                      ColorSequenceMixin, LegendMixin, ToolMixin, AnnotationMixin,
                      ToolboxMixin):
    """
    Wrapper for a :code:`bokeh.plotting.figure` object

    - Handles initialization of plots, glpyhs etc
    - Handles figure-level elements (axis labels, size/shape)

    This is a base class which doesn't strictly rely on Panel,
    it could be used in a pure Bokeh environment if you insert
    self.fig into the current Bokeh document. The child :class:`~BokehFigure`
    which inherits this base class is a full Panel component.
    """
    def __init__(self, **kwargs):
        """
        Create the figure object

        By default this class uses a 'minimum' sizing policy,
        which means the figure will take up as little space as possible
        after any fixed-sizing values are taken into account.

        Parameters:
        -----------
        **kwargs, dict, optional
            Extra keyword arguments which are passed directly to
            :code:`bokeh.plotting.figure`
        """
        width_policy = kwargs.pop('width_policy', 'min')
        height_policy = kwargs.pop('height_policy', 'min')
        x_range = kwargs.pop('x_range', self._get_range1d(start=0., end=1.))
        y_range = kwargs.pop('y_range', self._get_range1d(start=0., end=1.))
        self._fig = figure(x_range=x_range, y_range=y_range, **kwargs)
        self._fig.width_policy = width_policy
        self._fig.height_policy = height_policy
        self._fig.toolbar.active_inspect = None
        self._is_image_fig = False
        self._renderers = {}
        self._tools = {}
        self._doc = self.get_doc()

    @property
    def fig(self) -> Figure:
        """
        Get the current Bokeh figure

        Returns
        -------
        figure
            The Bokeh figure object
        """
        return self._fig

    @contextlib.contextmanager
    def as_parent(self, element: GlyphBase):
        """
        Context manager which allows a GlyphBase to register itself
        as 'parent' of another GlyphBase, such that removing the parent
        will also remove the renderers belonging to the child

        :meta private:
        """
        self.parent_elements.append(element)
        yield
        pop_from_list(self.parent_elements, element)

    @property
    def parent_elements(self) -> list[GlyphBase]:
        """
        A semaphore-like list which is set by :code:`self.as_parent`
        to track if we are creating an element on the figure from
        another element. This is needed to build the correct mapping
        from elements to renderers, so that a multi-renderer element
        can be removed from the figure completely.

        Returns
        -------
        list[GlyphBase]
            elements which are parents of the element
            being currently  created


        :meta private:
        """
        try:
            return self._parent_elements
        except AttributeError:
            self._parent_elements = []
            return self.parent_elements

    def add_layout(self, renderer: GlyphRenderer, position: str, insert: int | None = None):
        """
        Wraps Figure.add_layout

        Parameters
        ----------
        renderer : GlyphRenderer
            The renderer to add to the figure
        position : str
            The position to add the renderer (see Bokeh place spec)
        insert : Optional[int], optional
            Where to insert the renderer, by default None in which case standard
            .append is used.


        :meta private:
        """
        if insert is None or (not getattr(self.fig, position)):
            # Appending or a new figure layout (empty list)
            self.fig.add_layout(renderer, position)
        else:
            assert isinstance(insert, int)
            # Insert at specified position, will append or prepend
            # if insert_at is a nonsensical location
            getattr(self.fig, position).insert(insert, renderer)

    def add_glyph(self,
                  element: GlyphBase,
                  source: ColumnDataSource,
                  glyph: Glyph,
                  **kwargs) -> GlyphRenderer:
        """
        Add a glyph from element onto the figure using data from source

        Parameters
        ----------
        element : GlyphBase
            The element which owns the glyph
        source : ColumnDataSource
            The datasource to supply the renderer
        glyph : Glyph
            The glyph to add

        Returns
        -------
        GlyphRenderer
            The renderer of the glyph on the figure


        :meta private:
        """
        renderer = self.fig.add_glyph(source, glyph, **kwargs)
        for el in [element] + self.parent_elements:
            try:
                self._renderers[el].append(renderer)
            except KeyError:
                self._renderers[el] = [renderer]
        return renderer

    @property
    def elements(self) -> dict[str, GlyphBase]:
        return {el.name: el for el in self._renderers.keys()}

    def __getitem__(self, name: str) -> GlyphBase:
        return self.elements[name]

    def __setitem__(self, name: str, value: GlyphBase):
        raise NotImplementedError('Cannot add GlyphBases by name at this time')

    def get_renderers(self,
                      element: GlyphBase,
                      filter_glyphs: tuple[type[Glyph]] = None) -> list[GlyphRenderer]:
        """
        Get renderers linked to the element, optionally filtered to a particular glyph type

        Parameters
        ----------
        element : GlyphBase
            The element to get renderers for
        filter_glyphs : tuple[Type[Glyph]], optional
            Optional tuple of glyph tyes to filter by, by default None

        Returns
        -------
        list[GlyphRenderer]
            Renderes linked to the element


        :meta private:
        """
        if filter_glyphs is None:
            filter_glyphs = (Glyph,)
        return [r for r in self._renderers.get(element, [])
                if isinstance(r.glyph, filter_glyphs)]

    def remove_element(self, element: GlyphBase):
        """
        Remove all renderers associated with a GlyphBase
        Removes renderers on parent/child GlyphBases, also

        :meta private:
        """
        for renderer in self._renderers.pop(element, []):
            pop_from_list(self.fig.renderers, renderer)
            # Handle other references to renderer
            for _rs in self._renderers.values():
                pop_from_list(_rs, renderer)
        self._renderers = {k: v for k, v in self._renderers.items() if v}
        # Need to pop from self._tools ?
        # Need to pop from LegendItems ?

    def remove_renderer(self, renderer: GlyphRenderer):
        """
        Remove a specific renderer from the figure and update
        our mapping of self._renderers

        :meta private:
        """
        pop_from_list(self.fig.renderers, renderer)
        # Handle other references to renderer
        for _rs in self._renderers.values():
            pop_from_list(_rs, renderer)
        self._renderers = {k: v for k, v in self._renderers.items() if v}

    @property
    def doc(self) -> Document | None:
        """
        Get the current Bokeh document linked to this figure, if set
        by the :meth:`BokehFigureBase.set_doc` context manager.

        Returns
        -------
        Optional[Document]
            The document, if set


        :meta private:
        """
        try:
            return self._doc
        except AttributeError:
            return None

    def get_doc(self) -> Document:
        """
        Get the document which Bokeh considers to be the current
        document. This is not necessarily the correct one to update
        depending on the use of panel / threads / async.

        Returns
        -------
        Document
            A Bokeh document which might be the current one


        :meta private:
        """
        doc = self.doc
        if doc is None:
            doc = curdoc()
        return doc

    @contextlib.contextmanager
    def set_doc(self, doc: Document = None):
        """
        Set the Bokeh document which should be used for
        updates to the plot. Used when managing async callbacks
        and threads. Depending on the context this could come from
        Bokeh curdoc() or pn.state.curdoc

        Parameters
        ----------
        doc : Document, optional
            The correct document to update. If doc is None, will
            try to use :code:`self.get_doc()` to set the document.

        :meta private:
        """
        if doc is None:
            doc = self.get_doc()
        if doc is not None:
            self._doc = doc

[docs]class BokehFigure(BokehFigureBase, pn.pane.Bokeh):
    """
    :class:`~BokehFigureBase` which is a standalone Panel component

    This class subclasses pn.pane.Bokeh which means it is treated
    like a standard Panel component able to be inserted into layouts
    and also displayed directly in a notebook or made .servable()

    As a result it is also a param.Parametrized class with default
    Parameters which come from Panel. Currently these parameters are
    not taken into account or set by the BokehFigure class, but this
    could be harmonized in the future with a bit of refactoring.
    """
    # Priority is an internal Panel variable which determines precedence
    # for converting an object into a Panel pane. Setting 0.4 (which is
    # lower than the baseclass default) enures that pn.pane.Bokeh (priority 0.8)
    # remains the default choice for normal Bokeh objects like widgets and
    # other figures (rather than them being wrapped in an instance of this class)
    priority = 0.4

    def __init__(self, name=None, pane_kwargs=None, **kwargs):
        # Unpack the name kwarg into both Figure and pn.pane.Bokeh kwargs
        # This maintains a bit of consistency for referencing figures
        # By default Panel gives the pane a name, but Bokeh's default is None
        pane_kwargs = {} if pane_kwargs is None else pane_kwargs
        if isinstance(name, str):
            pane_kwargs['name'] = name
            kwargs['name'] = name
        # Directly specifying __init__ because we need to set object=self._fig
        # I think this could be made automatic with better inheritance design
        BokehFigureBase.__init__(self, **kwargs)
        pn.pane.Bokeh.__init__(self, object=self._fig, **pane_kwargs)
        pn.state.onload(self.refresh_pane)


[docs]    def refresh_pane(self):
        """
        Trigger an update of the panel pane displaying the Bokeh figure

        This is sometimes necessary if callbacks are not correctly updating
        the figure, particularly in the Jupyter context.
        """
        for el in self._renderers.keys():
            if isinstance(el, BokehImage):
                el.refresh_display()
        self.param.trigger('object')



[docs]    def update_pane(self):
        """
        Trigger an update to the panel pane, specifically tailored for
        a change to the figure size. This is probably not necessay
        as :meth:`~BokehFigure.refresh_pane` should do the same job.
        """
        spacer_size = 100
        displayed_size = self.get_displayed_size()
        spacer = Spacer()
        spacer.height_policy = 'fixed'
        spacer.width_policy = 'fixed'
        spacer.height = displayed_size.get('displayed_height', spacer_size)
        spacer.width = displayed_size.get('displayed_width', spacer_size)
        self.object = spacer
        self.object = self.fig


    def get_doc(self):
        """
        Get the document which panel considers to be the current
        document. This is not necessarily the correct one to update
        depending on the use of panel / threads / async.

        Returns
        -------
        Document
            A Bokeh document which might be the current one


        :meta private:
        """
        doc = self.doc
        if doc is None:
            doc = pn.state.curdoc
        return doc

    def set_visible(self, state: bool, *renderers: GlyphRenderer):
        doc = self.get_doc()
        to_update = []
        for ri, r in enumerate(renderers):
            if r.visible != state:
                if doc is None:
                    r.visible = state
                else:
                    to_update.append(ri)
        if doc:
            doc.add_next_tick_callback(lambda: [renderers[ri].update(visible=state)
                                                for ri in to_update])
