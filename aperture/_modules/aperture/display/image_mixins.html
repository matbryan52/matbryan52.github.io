                     title: str | None = None,
                     palette: str | None = None,
                     location: str = 'auto',
                     dimension: int = 25,
                     **kwargs):
        """
        Adds a colorbar to the image

        Parameters
        ----------
        title : Optional[str], optional
            A title/label to add to the colorbar, by default None,
            in which case the image array attributes are checked for
            an 'array_unit' which will be used if present
        palette : Optional[str], optional
            A Bokeh color palette string, by default None in which
            case a default palette will be used (greyscale)
        location : str, optional
            A Bokeh location specification string, by default 'auto'
        dimension: int, optional
            The width/height of the colorscale (depending on orientation),
            by default 25 pixels. Sets the narrow dimension as appropriate.
        **kwargs : dict, optional
            Optional :code:`kwargs` passed to the Bokeh ColorBar constructor, which
            can be used to set style and other options.
        """
        if location == 'auto':
            toolbar_location = self.fig._toolbar_placement()
            location = 'above' if toolbar_location == 'right' else 'right'
        if title is None and 'array_unit' in self.display_image.attrs:
            title = f'{self.display_image.attrs["array_unit"]}'
        if palette is not None:
            self.set_color_mapper(palette=palette)
        self._build_colorbar(title=title, **kwargs)
        self._add_colorbar(location, short_dimension=dimension)

[docs]    def link_colormap(self: BokehImage, bkim: BokehImage):
        """
        Links the colormap of this image to that of another image

        Configures callbacks such that only wigets on the parent image
        bkim will control the colormap.

        Parameters
        ----------
        bkim : BokehImage
            The other image with a colormap/bar
        """
        assert isinstance(bkim, self.__class__)
        assert bkim.color_mapper is not None
        self.set_color_mapper(mapper=bkim.color_mapper, callback=False)
        bkim._add_linked_image(self)
        cb = CustomJS(args={'target_cds': bkim.display_meta_cds},
                      code=self._propagate_clims_js(),)
        self.display_meta_cds.js_on_change('data', cb)


    @staticmethod
    def _propagate_clims_js():
        return '''
target_cds.data.low[0] = Math.min(target_cds.data._low[0], cb_obj.data._low[0])
target_cds.data.high[0] = Math.max(target_cds.data._high[0], cb_obj.data._high[0])
target_cds.change.emit()
target_cds.data = target_cds.data;
target_cds.properties.data.change.emit();
'''

    def _add_linked_image(self: BokehImage, child_image: BokehImage):
        try:
            self._child_images.append(child_image)
        except AttributeError:
            self._child_images = [child_image]

    def _combined_minmax(self: BokehImage) -> tuple[float, float]:
        try:
            minmax = [i._disp_minmax for i in self._child_images + [self]]
            allmin = min(v[0] for v in minmax)
            allmax = max(v[1] for v in minmax)
            return (allmin, allmax)
        except AttributeError:
            return self._disp_minmax

    def change_cmap(self: BokehImage, palette: str):
        if self.cmap_select:
            self.cmap_select.value = cmaps.get_colormap(palette, inverted=self.is_cmap_inverted())
        elif self.color_mapper:
            self.change_cmap_cb(None, palette=palette)

            pass

    def change_cmap_cb(self: BokehImage, event, palette: str | None = None):
        """
        Callback triggered by changing the colormap select box

        :meta private:
        """
        if palette is None:
            palette = event.new
        if self.color_mapper is not None:
            self._current_palette_name = palette
            # can definitely be JS-linked!!!
            self.color_mapper.palette = cmaps.get_colormap(palette,
                                                           inverted=self.is_cmap_inverted())

    def invert_cmap(self: BokehImage, event):
        if self.cmap_select is not None:
            self.cmap_select.param.trigger('value')

[docs]    def set_max_clim_manual(self: BokehImage, low: float, high: float, force=False):
        """
        Force a particular low/high limit for the colormapper
        To maintain this over updates to the displayed image
        supply fix_clims=True when calling .update_raw_image

        Updates both the colormap and the colormap slider, if present,
        with the new lower/upper values.

        Parameters
        ----------
        low : float
            Low-limit for the colormap
        high : float
            High-limit for the colormap
        """
        assert high > low
        self.display_minmax = (low, high)
        if force or pn.state.curdoc is None:
            if self.color_mapper is not None:
                self.color_mapper.low = low
                self.color_mapper.high = high
            if self.cbar_slider is not None:
                self.cbar_slider.start = low
                self.cbar_slider.end = high
                self.cbar_slider.value = (low, high)
                self.cbar_slider.step = slider_step_size(low, high)

[docs]    def get_cbar_slider(self: BokehImage,
                        title: str = 'Colormap range',
                        init_range: tuple[float, float] | None = None,
                        **kwargs) -> RangeSlider:
        """
        Create a two-handled RangeSlider used to interactively modify
        the colormapper lower/upper limits.

        Parameters
        ----------
        title : str, optional
            A label to apply to the slider, by default 'Colormap range'
        fixed_range : Optional[tuple[float, float]], optional
            Explicit initial range for the slider, by default None, in which
            case the current image min/max will be used.
        **kwargs : dict, optional
            Optional :code:`kwargs` passed to the RangeSlider constructor, which
            can be used to set style and other options.

        Returns
        -------
        RangeSlider
            The colormap slider with callbacks registered.
        """
        if self.has_cbar_slider():
            return self.cbar_slider

        if not init_range:
            init_range = self.display_minmax

        nstep = 300
        self._cbar_slider = RangeSlider(title=title,
                                        start=init_range[0],
                                        end=init_range[1],
                                        value=init_range,
                                        step=slider_step_size(*init_range, n=nstep),
                                        syncable=False,
                                        **kwargs)

        clim_value_callback = CustomJS(args={'cmapper': self.color_mapper,
                                             'cds': self.display_meta_cds},
                                       code=self._clim_slider_value_js())
        self._cbar_slider.js_on_change('value', clim_value_callback)

        self.display_meta_cds.data.update({'cbar_slider': [True]})
        clim_update_callback = CustomJS(args={'clim_slider': self._cbar_slider,
                                              'cmapper': self.color_mapper,
                                              'nstep': nstep},
                                        code=self._clim_slider_update_image_js())
        self.display_meta_cds.js_on_change('data', clim_update_callback)
        return self.cbar_slider


    @staticmethod
    def _clim_slider_update_image_js():
        return '''
var low = cb_obj.data.low[0]
var high = cb_obj.data.high[0]

clim_slider.start = low
clim_slider.end = high
clim_slider.value = [low, high];
clim_slider.step = (high - low) / nstep;

if (cb_obj.data.cbar_centered[0]){
    const val = Math.max(Math.abs(low), Math.abs(high))
    low = -val
    high = val
}

cmapper.low = low;
cmapper.high = high;
'''

    @staticmethod
    def _clim_slider_value_js():
        return '''
var low = cb_obj.value[0]
var high = cb_obj.value[1]

if (cds.data.cbar_centered[0]){
    const val = Math.max(Math.abs(low), Math.abs(high))
    low = -val
    high = val
}

cmapper.low = low;
cmapper.high = high;
'''

[docs]    def set_color_mapper(self: BokehImage,
                         palette: str | None = None,
                         mapper: ColorMapper | None = None,
                         callback: bool = True):
        """
        Set the colormap for the image. Can supply either a palette
        name, in which case a linear colormapper will be created, or an
        already initialized Bokeh ColorMapper which will be linked
        to the image and colorbar, if present.

        Must test how this behaves wrt. callbacks if this is called
        after the sliders/selectors have been initialized!!!

        Parameters
        ----------
        palette : Optional[str], optional
            A colormap palette name, by default None, in which case a default
            palette will be used.
        mapper : ColorMapper, optional
            A bokeh ColorMapper sub-class, by default None, overrides palette
            if both arguments are suppled.
        """
        if mapper is None:
            disprange = self.display_minmax
            if disprange[0] == disprange[1]:
                offset = 0.1
                disprange = (disprange[0] - offset, disprange[0] + offset)
            if palette is None:
                palette = self.current_palette_name
            else:
                self._current_palette_name = palette
            palette = cmaps.get_colormap(palette)
            self._color_mapper = LinearColorMapper(palette=palette,
                                                   low=disprange[0],
                                                   high=disprange[1],
                                                   syncable=False)
        else:
            self._color_mapper = mapper

        if callback:
            mapper_cb = CustomJS(code=self._colormap_limit_js(),
                                 args={'mapper': self._color_mapper})
            self.display_meta_cds.js_on_change('data', mapper_cb)

        self.im.glyph.color_mapper = self.color_mapper
        if self.color_bar is not None:
            self.color_bar.color_mapper = self.color_mapper


    @staticmethod
    def _colormap_limit_js():
        return '''
if (!(cb_obj.data.cbar_slider[0])){
    var low = cb_obj.data.low[0]
    var high = cb_obj.data.high[0]
    if (cb_obj.data.cbar_centered[0]){
        const val = Math.max(Math.abs(low), Math.abs(high))
        low = -val
        high = val
    }
    mapper.low = low
    mapper.high = high
}
'''

[docs]    def add_hover(self: BokehImage, **kwargs):
        """
        Adds a hover tool which can be used to inspect pixel values
        and coordinates in the image array

        Parameters
        ----------
        **kwargs : dict, optional
            Optional :code:`kwargs` passed directly to the Bokeh HoverTool
            constructor.
        """
        # Can perhaps be refactored into a mixin with XYPlot
        label = self.hover_label()
        if not self.fig.has_tool(label):
            tool = self._make_hover(mode='mouse',
                                    description='Inspect value of image array',
                                    tooltips=self._hover_tooltips(),
                                    **kwargs)
            self.fig.add_tool(label, tool)
        self.fig.add_renderers_to_tool(label,
                                       self.fig.get_renderers(self,
                                                              filter_glyphs=(Image,)))


    @staticmethod
    def hover_label() -> str:
        # internal name for this tool when registering on figure
        return 'image_hover'

                         alpha: float | None = None,
                         name: str = 'Alpha',
                         **kwargs) -> Slider:
        """
        Get a slider allowing setting the image alpha value interactively

        Parameters
        ----------
        alpha : Optional[float], optional
            An initial alpha value, by default None in which case 1.0
        name : str, optional
            A label to apply  to the slider, by default 'Alpha'
        **kwargs : dict, optional
            Optional :code:`kwargs` passed to the Slider constructor, which
            can be used to set style and other options.

        Returns
        -------
        Slider
            The alpha slider
        """
        if not self.alpha_slider:
            self._alpha_slider = self._make_alpha_slider(alpha=alpha, name=name, **kwargs)
            self.link_alpha_slider(self._alpha_slider)
        return self.alpha_slider


    def link_alpha_slider(self: BokehImage, slider: Slider):
        callback = CustomJS(args={'glyph': self._glyph}, code=self._set_alpha_js())
        slider.js_on_change('value', callback)

    @staticmethod
    def _set_alpha_js():
        return """
glyph.global_alpha = cb_obj.value;
"""

    def clamp_alpha(self, *args):
        current = self.current_alpha()
        new = min(1., max(0., abs(1. - round(current))))
        self.set_alpha(new)

    def alpha_toggle(self: BokehImage):
        try:
            return self._alpha_toggle
        except AttributeError:
            return None

    def get_alpha_toggle(self: BokehImage,
                         name: str = 'Toggle alpha',

        if not self.alpha_toggle:
            self._alpha_toggle = pn.widgets.Button(name=name,
                                                   width=kwargs.pop('width', 150),
                                                   **kwargs)
            self._alpha_toggle.on_click(self.clamp_alpha)
        return self.alpha_toggle
