class RunUDF_UIBase(PickUDF_UI):
    _label = 'Run'

    @classmethod
    def base_panel(cls, ctx: lt.Context, ds: lt.DataSet, page_title=True,
                   initialize=True, pick_results=None, layout=None):
        """
        Builds the base Pick UDF UI panel and adds a tool to draw rectangle masks
        on the navigation image. These rectangles will be used to set ctx.run_udf(..., roi)
        if the method on this class are used to run the imaging callback

        :meta private:
        """
        if layout is None:
            layout = AutoTwoPane()
        if page_title:
            layout.add_title(title=page_title
                             if isinstance(page_title, str)
                             else f'{cls._label} UDF')
        cls.pick_function(layout, ctx, ds,
                          initialize=initialize,
                          results_container=pick_results)
        nav_fig: BokehFigure = layout.panes[0]['nav']
        nav_fig.add_rectangles(allow_edit=True,
                               name='rectangles',
                               line_color=cls._default_cursor_color,
                               fill_color=cls._default_cursor_color)
        nav_fig.add_polygons(allow_edit=True,
                             name='polygons',
                             line_color=cls._default_cursor_color,
                             fill_color=cls._default_cursor_color)
        nav_fig.add_annotators('rectangles', 'polygons')
        return layout

    @classmethod
    def setup_run(cls,
                  ctx: lt.Context,
                  ds: lt.DataSet,
                  layout: LayoutContainer,
                  build_udfs_cb: Callable[[CallbackContainer], list[UDF]],
                  cb_obj: CallbackContainer,
                  roi_cb: Callable = None,
                  **run_widgets_kwargs):
        """
        Setup callbacks to run imaging on button press

        Parameters
        ----------
        ctx : lt.Context
            The context to use
        ds : lt.DataSet
            The dataset to run on
        layout : pn.layout.ListPanel
            The layout on which to add the run/stop/progress widgets
        build_udfs_cb : Callable[[CallbackContainer], list['UDF']]
            A callback to build the UDF object to run, which should be
            performs the necessary unpacking of cursors, LiberTEM analyses
            etc to return a list of UDFs. The cb_obj is provided so that
            this function can optionally set state on the container if
            this is useful for the display part of the callback
        cb_obj : CallbackContainer
            Container object passed to the callback to configure
            what to do with results, which elements to disable/enable etc

        :meta private:
        """
        cls.add_run_widgets(layout, cb_obj, **run_widgets_kwargs)
        if roi_cb is None:
            nav_fig: BokehFigure = layout['nav']
            roi_cb = cls.get_roi_cb(nav_fig, ds.shape.nav)
        _run_cb = cls.build_async_iter_callback(cb_obj,
                                                ctx,
                                                ds,
                                                build_udfs_cb,
                                                roi_cb)
        layout['run_button'].on_click(_run_cb)

    @staticmethod
    def get_roi_cb(fig, shape):
        return partial(fig.annotations.get_mask, shape=shape)


class Imaging_UIBase(RunUDF_UIBase):
    """
    Provides convenience methods on top of the PickUDF functionality
    specific to virtual imaging-like features

    :meta private:
    """

    _label = 'Imaging'

    @staticmethod
    def initial_radii(ds):
        r0 = max(ds.shape.sig) / 5.
        r1 = max(ds.shape.sig) / 3.
        max_radius = max(ds.shape.sig) / 1.5
        return r0, r1, max_radius

    @classmethod
    def add_switchable_cursor(cls,
                              layout_select: LayoutContainer,
                              layout_slider: LayoutContainer,
                              ds,
                              img: BokehImage,
                              cselect_kwargs=None,
                              active=True) -> Cursor:

        fig = img.fig
        annotation_kwargs = {'line_color': cls._default_cursor_color,
                             'fill_color': cls._default_cursor_color,
                             'fill_alpha': 0.3}

        r0, r1, max_radius = cls.initial_radii(ds)
        cursor = fig.add_switchable_cursor(image=img,
                                           name='cursor',
                                           line_color=cls._default_cursor_color,
                                           annotation_kwargs=annotation_kwargs)
        if active:
            fig.set_active_tool(cursor.tool_label())
        if cselect_kwargs is None:
            cselect_kwargs = {}
        cursor_select = cursor.make_cursor_select(**cselect_kwargs)
        layout_select.add_element('cursor_select', cursor_select)
        radius_slider = cursor.make_radius_slider(r0=r0, r1=r1, max_r=max_radius)
        layout_slider.add_element('radius_slider', radius_slider)
        return cursor

    @classmethod
    def setup_imaging(cls, *args, **kwargs):
        return cls.setup_run(*args, **kwargs)



[docs]class ImagingCallbackContainer(CallbackContainer):
    @property
    def results_image(self) -> BokehImage:
        return self._results_image

    @results_image.setter
    def results_image(self, val: BokehImage):
        self._results_image = val

    @staticmethod
    def buffer_unpack(buffers):
        return buffers['intensity'].data.squeeze(axis=-1)

    def partial_update(self, part_results):
        array = self.buffer_unpack(part_results.buffers[0])
        self.results_image.update_raw_image(array)




[docs]class Imaging_UI(Imaging_UIBase):
    """
    Create a UI panel to perform virtual imaging on a LiberTEM dataset

    The user can interactively browse the diffraction patterns (a Pick UDF
    is automatically run each time the user interacts), configure the
    imaging (mask type, radii, centre etc.) then launch the analysis
    with iterative updating and a progress bar. The analysis runs
    asynchronously so the UI should remain responsive during computation.

    Supports point-, disk- and ring- LiberTEM mask analyses, and automatically
    adds functionality to draw one-or-more rectangle ROIs on the navigation pane
    """
    _label = 'Imaging'


[docs]    @classmethod
    def panel(cls,
              ctx: lt.Context,
              ds: lt.DataSet,
              results_container: ResultTree | None = None,
              page_title: bool | str = True) -> AutoTwoPane:
        """
        Build the UI for the specific dataset and setup the analyses
        using the suplied context

        The returned layout object can be served if running with
        :code:`panel serve`, interactively launched using :code:`.show()`
        or displayed directly in a notebook by returning it from a cell

        Parameters
        ----------
        ctx : lt.Context
            LiberTEM context object
        ds : lt.DataSet
            LiberTEM dataset object
        results_container : Optional[ResultTree]
            A :class:`.ResultTree` object
            which the UI will fill with results and metadata from
            the analysis run
        page_title : Union[bool, str]
            If a string, set this as the layout title, else use a
            default title if True. Set no page title if False.

        Returns
        -------
        TwoPane
            The UI panel
        """
        if results_container is None:
            results_container = ResultTree()

        results_container.set_schema(cls.imaging_result_schema)
        pick_results = results_container.add_child('pick')

        layout = cls.base_panel(ctx, ds, page_title=page_title, pick_results=pick_results)
        cselect_kwargs = {'title': 'Imaging Mode',
                          'mapping': {'Virual Imaging': 'point',
                                      'Brightfield / Disk': 'disk',
                                      'ADF / Ring': 'annulus'},
                          'value': 'Brightfield / Disk'}
        cls.add_switchable_cursor(layout.panes[1],
                                  layout.panes[1],
                                  ds,
                                  layout.panes[1]['sig']['img'],
                                  cselect_kwargs=cselect_kwargs)
        cursor = layout.panes[1]['sig']['cursor']

        def build_udfs(cb_obj):
            parameters = cursor.get_params()
            param_mapping = {'cx': 'cx', 'cy': 'cy'}
            cx, cy = parameters['cx'], parameters['cy']
            if cx and cy:
                position_stub = f'xy{cx:.1f},{cy:.1f}'
                if parameters['mode'] == 'Virual Imaging':
                    analysis_params = {v: parameters.get(k) for k, v in param_mapping.items()}
                    analysis = PointMaskAnalysis(ds, analysis_params)
                    decriptor = f'virtual_detector_{position_stub}'
                elif parameters['mode'] == 'Brightfield / Disk':
                    param_mapping = {**param_mapping, 'r0': 'r'}
                    analysis_params = {v: parameters.get(k) for k, v in param_mapping.items()}
                    analysis = DiskMaskAnalysis(ds, analysis_params)
                    r0 = parameters['r0']
                    decriptor = f'disksum_{position_stub}_r{r0:.1f}'
                elif parameters['mode'] == 'ADF / Ring':
                    param_mapping = {**param_mapping, 'r0': 'ri', 'r1': 'ro'}
                    analysis_params = {v: parameters.get(k) for k, v in param_mapping.items()}
                    analysis = RingMaskAnalysis(ds, analysis_params)
                    r0, r1 = parameters['r0'], parameters['r1']
                    decriptor = f'ringsum_{position_stub}_r01{r0:.1f},{r1:.1f}'
                else:
                    raise RuntimeError(f'Unavailable mode: {parameters["mode"]}')
                udfs = [analysis.get_udf()]
                meta = {**analysis_params, 'mode': parameters['mode'], 'descriptor': decriptor}
                return udfs, meta
            return [], {}

        nav_fig = layout.panes[0]['nav']
        nav_im: BokehImage = nav_fig['img']
        layout.panes[0].add_element('nav_img_toolbox', nav_fig.get_toolbox(name='Nav toolbox'))
        layout.panes[1].add_element('sig_img_toolbox',
                                    layout.panes[1]['sig'].get_toolbox(name='Sig toolbox'))

        cb_obj = ImagingCallbackContainer(layout, results_container=results_container)
        cb_obj.results_image = nav_im
        cls.setup_imaging(ctx, ds, layout.panes[0],
                          build_udfs, cb_obj)
        return layout.finalize()

            }
